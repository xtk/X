 44 /**
 45  * Create a parser for the binary .CRV format of Freesurfer.
 46  * 
 47  * @constructor
 48  * @extends X.parser
 49  */
 50 X.parserCRV = function() {
 51 
 52   //
 53   // call the standard constructor of X.base
 54   goog.base(this);
 55   
 56   //
 57   // class attributes
 58   
 59   /**
 60    * @inheritDoc
 61    * @const
 62    */
 63   this['_className'] = 'parserCRV';
 64   
 65 };
 66 // inherit from X.parser
 67 goog.inherits(X.parserCRV, X.parser);
 68 
 69 
 70 /**
 71  * @inheritDoc
 72  */
 73 X.parserCRV.prototype.parse = function(object, data) {
 74 
 75   var ind = object._pointIndices;
 76   
 77   // we need point indices here, so fail if there aren't any
 78   if (ind.length == 0) {
 79     
 80     throw new Error('No _pointIndices defined on the X.object.');
 81     
 82   }
 83   
 84   var currentOffset = 0;
 85   
 86   var magicNumber = this.parseUInt24EndianSwapped(data, currentOffset);
 87   currentOffset += 3;
 88   
 89   // This hackery is the fact that the new version defines this
 90   // as a magic number to identify the new file type
 91   if (magicNumber != 16777215) {
 92     alert("Can't load curvature file, invalid magic number.");
 93     return;
 94   }
 95   
 96   var numVertices = this.parseUInt32EndianSwapped(data, currentOffset);
 97   
 98   currentOffset += 4;
 99   var fnum = this.parseUInt32EndianSwapped(data, currentOffset);
100   currentOffset += 4;
101   var valsPerVertex = this.parseUInt32EndianSwapped(data, currentOffset);
102   currentOffset += 4;
103   
104   var numPosValues = 0;
105   var numNegValues = 0;
106   var negSum = 0.0;
107   var posSum = 0.0;
108   var posMean = 0.0;
109   var negMean = 0.0;
110   var posStdDev = 0.0;
111   var negStdDev = 0.0;
112   var sum = 0.0;
113   var mean = 0.0;
114   var stdDev = 0.0;
115   var numValues = 0;
116   var vertexCurvatures = new Array(numVertices);
117   var minCurv = new Array(2);
118   var maxCurv = new Array(2);
119   
120   var k;
121   for (k = 0; k < numVertices; k++) {
122     var curv = this.parseFloat32EndianSwapped(data, currentOffset);
123     currentOffset += 4;
124     if (k == 0) {
125       minCurv[0] = maxCurv[0] = curv;
126     }
127     if (curv >= 0.0) {
128       numPosValues++;
129       posSum += curv;
130     } else {
131       numNegValues++;
132       negSum += curv;
133     }
134     
135     sum += curv;
136     numValues++;
137     
138     maxCurv[0] = Math.max(curv, maxCurv[0]);
139     minCurv[0] = Math.min(curv, minCurv[0]);
140     
141     vertexCurvatures[k] = curv;
142   }
143   
144   if (numPosValues != 0) {
145     posMean = posSum / numPosValues;
146   }
147   
148   if (numNegValues != 0) {
149     negMean = negSum / numNegValues;
150   }
151   
152   if (numValues != 0) {
153     mean = sum / numValues;
154   }
155   
156 
157   posSum = 0.0;
158   negSum = 0.0;
159   sum = 0.0;
160   
161   var i;
162   for (i = 0; i < numVertices; i++) {
163     var curv = vertexCurvatures[i];
164     var diffSq = 0;
165     if (curv >= 0.0) {
166       diffSq = Math.pow((curv - posMean), 2);
167       posSum += diffSq;
168     } else {
169       diffSq = Math.pow((curv - negMean), 2);
170       negSum += diffSq;
171     }
172     
173     diffSq = Math.pow((curv - mean), 2);
174     sum += diffSq;
175   }
176   
177   if (numPosValues > 1) {
178     posStdDev = Math.sqrt(posSum / (numPosValues - 1));
179   }
180   
181   if (numNegValues > 1) {
182     negStdDev = Math.sqrt(negSum / (numNegValues - 1));
183   }
184   
185   if (numValues > 1) {
186     stdDev = Math.sqrt(sum / (numValues - 1));
187   }
188   
189   // Store also 2.5 standard deviations from each mean. This is
190   // a more reasonable range to render with
191   minCurv[1] = negMean - 2.5 * negStdDev;
192   maxCurv[1] = posMean + 2.5 * posStdDev;
193   
194   //
195   // now order the curvature values based on the indices
196   //
197   var numberOfScalars = numVertices;
198   var numberOfIndices = ind.length;
199   
200   var orderedCurvatures = [];
201   
202   for (i = 0; i < numberOfIndices; i++) {
203     
204     var currentIndex = ind[i];
205     
206     // validate
207     if (currentIndex > numberOfScalars) {
208       
209       throw new Error('Could not find scalar for vertex.');
210       
211     }
212     
213     // grab the current scalar
214     var currentScalar = vertexCurvatures[currentIndex];
215     
216     // add the scalar 3x since we need to match the point array length
217     orderedCurvatures.push(currentScalar);
218     orderedCurvatures.push(currentScalar);
219     orderedCurvatures.push(currentScalar);
220     
221   }
222   
223   // attach min, max curvature values and the whole shebang!
224   object._scalars._min = minCurv[1];
225   object._scalars._max = maxCurv[1];
226   object._scalars['_minThreshold'] = minCurv[1];
227   object._scalars['_maxThreshold'] = maxCurv[1];
228   object._scalars._array = vertexCurvatures; // the un-ordered scalars
229   object._scalars.setGlArray(orderedCurvatures); // the ordered, gl-Ready
230   // version - use the setter to mark the scalars dirty
231   
232   var modifiedEvent = new X.event.ModifiedEvent();
233   modifiedEvent._object = object;
234   this.dispatchEvent(modifiedEvent);
235   
236 };
