 52 /**
 53  * Create an interactor for a given element in the DOM tree.
 54  * 
 55  * @constructor
 56  * @param {Element} element The DOM element to be observed.
 57  * @extends X.base
 58  */
 59 X.interactor = function(element) {
 60 
 61   // check if we have a valid element
 62   if (!goog.isDefAndNotNull(element) || !(element instanceof Element)) {
 63     
 64     throw new X.exception('Could not add interactor to the given element.');
 65     
 66   }
 67   
 68   //
 69   // call the standard constructor of X.base
 70   goog.base(this);
 71   
 72   //
 73   // class attributes
 74   
 75   /**
 76    * @inheritDoc
 77    * @const
 78    */
 79   this._className = 'interactor';
 80   
 81   /**
 82    * The observed DOM element of this interactor.
 83    * 
 84    * @type {!Element}
 85    * @protected
 86    */
 87   this._element = element;
 88   
 89   /**
 90    * The listener id for mouse wheel observation.
 91    * 
 92    * @type {?number}
 93    * @protected
 94    */
 95   this._mouseWheelListener = null;
 96   
 97   /**
 98    * The listener id for mouse down observation.
 99    * 
100    * @type {?number}
101    * @protected
102    */
103   this._mouseDownListener = null;
104   
105   /**
106    * The listener id for mouse up observation.
107    * 
108    * @type {?number}
109    * @protected
110    */
111   this._mouseUpListener = null;
112   
113   /**
114    * The listener id for mouse move observation.
115    * 
116    * @type {?number}
117    * @protected
118    */
119   this._mouseMoveListener = null;
120   
121   /**
122    * The listener id for mouse out observation.
123    * 
124    * @type {?number}
125    * @protected
126    */
127   this._mouseOutListener = null;
128   
129   /**
130    * The browser independent mouse wheel handler.
131    * 
132    * @type {?goog.events.MouseWheelHandler}
133    * @protected
134    */
135   this._mouseWheelHandler = null;
136   
137   /**
138    * Indicates if the mouse is inside the element.
139    * 
140    * @type {boolean}
141    * @protected
142    */
143   this._mouseInside = true;
144   
145   /**
146    * Indicates if the left mouse button is pressed.
147    * 
148    * @type {boolean}
149    * @protected
150    */
151   this._leftButtonDown = false;
152   
153   /**
154    * Indicates if the middle mouse button is pressed.
155    * 
156    * @type {boolean}
157    * @protected
158    */
159   this._middleButtonDown = false;
160   
161   /**
162    * Indicates if the right mouse button is pressed.
163    * 
164    * @type {boolean}
165    * @protected
166    */
167   this._rightButtonDown = false;
168   
169   /**
170    * The previous mouse position.
171    * 
172    * @type {!goog.math.Vec2}
173    * @protected
174    */
175   this._lastMousePosition = new goog.math.Vec2(0, 0);
176   
177 };
178 // inherit from X.base
179 goog.inherits(X.interactor, X.base);

184  *

187 X.interactor.prototype.config = {
188   MOUSEWHEEL_ENABLED: true,
189   MOUSECLICKS_ENABLED: true,
190   KEYBOARD_ENABLED: true,
191   HOVERING_ENABLED: true,
192   CONTEXTMENU_ENABLED: false

196 /**
197  * Observe mouse wheel interaction on the associated DOM element.
198  */
199 X.interactor.prototype.init = function() {
200 
201   if (this.config.MOUSEWHEEL_ENABLED) {
202     
203     // we use the goog.events.MouseWheelHandler for a browser-independent
204     // implementation
205     this._mouseWheelHandler = new goog.events.MouseWheelHandler(this._element);
206     
207     this._mouseWheelListener = goog.events.listen(this._mouseWheelHandler,
208         goog.events.MouseWheelHandler.EventType.MOUSEWHEEL, this.onMouseWheel
209             .bind(this));
210     
211   } else {
212     
213     // remove all mouse wheel observers, if they exist..
214     goog.events.unlistenByKey(this._mouseWheelListener);
215     
216     this._mouseWheelHandler = null;
217     
218   }
219   
220   if (this.config.MOUSECLICKS_ENABLED) {
221     
222     // mouse down
223     this._mouseDownListener = goog.events.listen(this._element,
224         goog.events.EventType.MOUSEDOWN, this.onMouseDown.bind(this));
225     
226     // mouse up
227     this._mouseUpListener = goog.events.listen(this._element,
228         goog.events.EventType.MOUSEUP, this.onMouseUp.bind(this));
229     
230   } else {
231     
232     // remove the observer, if it exists..
233     // goog.events.unlisten(this._element, goog.events.EventType.MOUSEDOWN);
234     goog.events.unlistenByKey(this._mouseDownListener);
235     
236     // remove the observer, if it exists..
237     goog.events.unlistenByKey(this._mouseUpListener);
238     
239   }
240   
241   if (!this.config.CONTEXTMENU_ENABLED) {
242     
243     // deactivate right-click context menu
244     // found no way to use goog.events for that? tried everything..
245     // according to http://help.dottoro.com/ljhwjsss.php, this method is
246     // compatible with all browsers but opera
247     this._element.oncontextmenu = function() {
248 
249       return false;
250       
251     };
252     
253   } else {
254     
255     // re-activate right-click context menu
256     this._element.oncontextmenu = null;
257   }
258   
259   if (this.config.KEYBOARD_ENABLED) {
260     
261     // the google closure way did not work, so let's do it this way..
262     window.onkeydown = this.onKey.bind(this);
263     
264   } else {
265     
266     // remove the keyboard observer
267     window.onkeydown = null;
268     
269   }
270   
271   //
272   // we always listen to mouse move events since they are essential for the
273   // other events
274   // we do make sure, we add them only once
275   
276   // remove the observer, if it exists..
277   goog.events.unlistenByKey(this._mouseMoveListener);
278   
279   // remove the observer, if it exists..
280   goog.events.unlistenByKey(this._mouseOutListener);
281   
282   // mouse movement inside the element
283   this._mouseMoveListener = goog.events.listen(this._element,
284       goog.events.EventType.MOUSEMOVE, this.onMouseMovementInside.bind(this));
285   
286   // mouse movement outside the element
287   this._mouseOutListener = goog.events.listen(this._element,
288       goog.events.EventType.MOUSEOUT, this.onMouseMovementOutside.bind(this));
289   
290 };

293 /**
294  * Callback for mouse down events on the associated DOM element.
295  * 
296  * @param {Event} event The browser fired event.
297  */
298 X.interactor.prototype.onMouseDown = function(event) {
299 
300   if (event.button == goog.events.BrowserEvent.MouseButton.LEFT) {
301     
302     // left button click
303     this._leftButtonDown = true;
304     
305   } else if (event.button == goog.events.BrowserEvent.MouseButton.MIDDLE) {
306     
307     // middle button click
308     this._middleButtonDown = true;
309     
310   } else if (event.button == goog.events.BrowserEvent.MouseButton.RIGHT) {
311     
312     // right button click
313     this._rightButtonDown = true;
314     
315   }
316   
317   // end all hovering since the scene can change and a caption might be
318   // misplaced etc.
319   this.hoverEnd_();
320   
321   // prevent further handling by the browser
322   event.preventDefault();
323   
324 };

327 /**
328  * Callback for mouse up events on the associated DOM element.
329  * 
330  * @param {Event} event The browser fired event.
331  */
332 X.interactor.prototype.onMouseUp = function(event) {
333 
334   if (event.button == goog.events.BrowserEvent.MouseButton.LEFT) {
335     
336     // left button click
337     this._leftButtonDown = false;
338     
339   } else if (event.button == goog.events.BrowserEvent.MouseButton.MIDDLE) {
340     
341     // middle button click
342     this._middleButtonDown = false;
343     
344   } else if (event.button == goog.events.BrowserEvent.MouseButton.RIGHT) {
345     
346     // right button click
347     this._rightButtonDown = false;
348     
349   }
350   
351   // end all hovering since the scene can change and a caption might be
352   // misplaced etc.
353   this.hoverEnd_();
354   
355   // prevent further handling by the browser
356   event.preventDefault();
357   
358 };

361 /**
362  * Callback for mouse movement events outside the associated DOM element. This
363  * resets all internal interactor flags.
364  * 
365  * @param {Event} event The browser fired event.
366  */
367 X.interactor.prototype.onMouseMovementOutside = function(event) {
368 
369   // reset the click flags
370   this._mouseInside = false;
371   if (this.config.KEYBOARD_ENABLED) {
372     
373     // if we observe the keyboard, remove the observer here
374     // this is necessary if there are more than one renderer in the document
375     window.onkeydown = null;
376     
377   }
378   
379   this._leftButtonDown = false;
380   this._middleButtonDown = false;
381   this._rightButtonDown = false;
382   // end all hovering since the scene can change and a caption might be
383   // misplaced etc.
384   this.hoverEnd_();
385   this._lastMousePosition = new goog.math.Vec2(0, 0);
386   
387   // prevent further handling by the browser
388   event.preventDefault();
389   
390 };

393 /**
394  * Callback for mouse movement events inside the associated DOM element. This
395  * distinguishes by pressed mouse buttons, key accelerators etc. and fires
396  * proper X.event events.
397  * 
398  * @param {Event} event The browser fired event.
399  */
400 X.interactor.prototype.onMouseMovementInside = function(event) {
401 
402   // TODO this needs to be more generalized
403   this.dispatchEvent('mouseup');
404   
405   this._mouseInside = true;
406   
407   if (this.config.KEYBOARD_ENABLED && window.onkeydown == null) {
408     
409     // we re-gained the focus, enable the keyboard observer again!
410     window.onkeydown = this.onKey.bind(this);
411     
412   }
413   
414   // prevent any other actions by the browser (f.e. scrolling, selection..)
415   event.preventDefault();
416   
417   // is shift down?
418   var shiftDown = event.shiftKey;
419   
420   // grab the current mouse position
421   var currentMousePosition = new goog.math.Vec2(event.offsetX, event.offsetY);
422   
423   // get the distance in terms of the last mouse move event
424   var distance = this._lastMousePosition.subtract(currentMousePosition);
425   
426   // save the current mouse position as the last one
427   this._lastMousePosition = currentMousePosition.clone();
428   
429   // 
430   // hovering, if enabled..
431   //
432   if (this.config.HOVERING_ENABLED) {
433     
434     if (Math.abs(distance.x) > 0 || Math.abs(distance.y) > 0 ||
435         this._middleButtonDown || this._leftButtonDown || this._rightButtonDown) {
436       
437       // there was some mouse movement, let's cancel the hovering countdown
438       this.hoverEnd_();
439       
440     }
441     
442     // start the hovering countdown
443     // if the mouse does not move for 2 secs, fire the HoverEvent to initiate
444     // picking etc.
445     this._hoverTrigger = setTimeout(function() {
446 
447       this.hoverEnd_();
448       
449       var e = new X.event.HoverEvent();
450       e._x = currentMousePosition.x;
451       e._y = currentMousePosition.y;
452       
453       this.dispatchEvent(e);
454       
455       // reset the trigger
456       this._hoverTrigger = null;
457       
458     }.bind(this), 700);
459     
460   }
461   
462   // threshold the distance to avoid 'irregular' movement
463   if (Math.abs(distance.x) < 2) {
464     
465     distance.x = 0;
466     
467   }
468   if (Math.abs(distance.y) < 2) {
469     
470     distance.y = 0;
471     
472   }
473   
474   // jump out if the distance is 0 to avoid unnecessary events
475   if (distance.magnitude() == 0) {
476     
477     return;
478     
479   }
480   
481 
482   //
483   // check which mouse buttons or keys are pressed
484   //
485   if (this._leftButtonDown && !shiftDown) {
486     //
487     // LEFT MOUSE BUTTON DOWN AND NOT SHIFT DOWN
488     //
489     
490     // create a new rotate event
491     var e = new X.event.RotateEvent();
492     
493     // attach the distance vector
494     e._distance = distance;
495     
496     // attach the angle in degrees
497     e._angle = 0;
498     
499     // .. fire the event
500     this.dispatchEvent(e);
501     
502 
503   } else if (this._middleButtonDown || (this._leftButtonDown && shiftDown)) {
504     //
505     // MIDDLE MOUSE BUTTON DOWN or LEFT MOUSE BUTTON AND SHIFT DOWN
506     //
507     
508     // create a new pan event
509     var e = new X.event.PanEvent();
510     
511     // panning in general moves pretty fast, so we threshold the distance
512     // additionally
513     if (distance.x > 5) {
514       
515       distance.x = 5;
516       
517     } else if (distance.x < -5) {
518       
519       distance.x = -5;
520       
521     }
522     if (distance.y > 5) {
523       
524       distance.y = 5;
525       
526     } else if (distance.y < -5) {
527       
528       distance.y = -5;
529       
530     }
531     
532     // attach the distance vector
533     e._distance = distance;
534     
535     // .. fire the event
536     this.dispatchEvent(e);
537     
538 
539   } else if (this._rightButtonDown) {
540     //
541     // RIGHT MOUSE BUTTON DOWN
542     //
543     
544     // create a new zoom event
545     var e = new X.event.ZoomEvent();
546     
547     // set the zoom direction
548     // true if zooming in, false if zooming out
549     e._in = (distance.y > 0);
550     
551     // with the right click, the zoom will happen rather
552     // fine than fast
553     e._fast = false;
554     
555     // .. fire the event
556     this.dispatchEvent(e);
557     
558 
559   }
560   
561 };

564 /**
565  * Stop the hover countdown and fire a X.event.HoverEndEvent.
566  */
567 X.interactor.prototype.hoverEnd_ = function() {
568 
569   if (this._hoverTrigger) {
570     clearTimeout(this._hoverTrigger);
571   }
572   
573   var e = new X.event.HoverEndEvent();
574   this.dispatchEvent(e);
575   
576 };

579 /**
580  * Callback for mouse wheel events on the associated DOM element. This fires
581  * proper X.event events.
582  * 
583  * @param {Event} event The browser fired event.
584  */
585 X.interactor.prototype.onMouseWheel = function(event) {
586 
587   // end all hovering since the scene can change and a caption might be
588   // misplaced etc.
589   this.hoverEnd_();
590   
591   // prevent any other action (like scrolling..)
592   event.preventDefault();
593   
594   // create a new zoom event
595   var e = new X.event.ZoomEvent();
596   
597   // make sure, deltaY is defined
598   if (!goog.isDefAndNotNull(event.deltaY)) {
599     event.deltaY = 0;
600   }
601   
602   // set the zoom direction
603   // true if zooming in, false if zooming out
604   // delta is here given by the event
605   e._in = (event.deltaY < 0);
606   
607   // with the mouseWheel, the zoom will happen rather
608   // fast than fine
609   e._fast = true;
610   
611   // .. fire the event
612   this.dispatchEvent(e);
613   
614 };

617 /**
618  * Callback for keyboard events on the associated DOM element. This fires proper
619  * X.event events.
620  * 
621  * @param {Event} event The browser fired event.
622  */
623 X.interactor.prototype.onKey = function(event) {
624 
625   // only listen to key events if the mouse is inside our element
626   // this f.e. enables key event listening for multiple renderers
627   if (!this._mouseInside) {
628     
629     return;
630     
631   }
632   
633   // end all hovering since the scene can change and a caption might be
634   // misplaced etc.
635   this.hoverEnd_();
636   
637   // observe the control keys (shift, alt, ..)
638   var alt = event.altKey;
639   var ctrl = event.ctrlKey;
640   var meta = event.metaKey; // this is f.e. the windows or apple key
641   var shift = event.shiftKey;
642   
643   // get the keyCode
644   var keyCode = event.keyCode;
645   
646   if (keyCode == 82 && !alt && !ctrl && !meta && !shift) {
647     
648     // 'r' but without any other control keys since we do not want to limit the
649     // user to press for example CTRL+R to reload the page
650     
651     // prevent any other actions..
652     event.preventDefault();
653     
654     // fire the ResetViewEvent
655     var e = new X.event.ResetViewEvent();
656     this.dispatchEvent(e);
657     
658   } else if (keyCode >= 37 && keyCode <= 40) {
659     
660     // keyCode <= 37 and >= 40 means the arrow keys
661     
662     // prevent any other actions..
663     event.preventDefault();
664     
665     var e = null;
666     
667     if (shift) {
668       
669       // create a new pan event
670       e = new X.event.PanEvent();
671       
672     } else if (alt) {
673       
674       // create a new zoom event
675       e = new X.event.ZoomEvent();
676       
677     } else {
678       // create a new rotate event
679       e = new X.event.RotateEvent();
680       
681     }
682     
683     if (!e) {
684       
685       // should not happen but you never know with key interaction
686       return;
687       
688     }
689     
690     // create a distance vector
691     var distance = new goog.math.Vec2(0, 0);
692     
693     if (keyCode == 37) {
694       // '<-' LEFT
695       distance.x = 5;
696       if (alt) {
697         // for zoom, we configure the zooming behavior
698         e._in = true;
699         e._fast = false;
700       }
701       
702     } else if (keyCode == 39) {
703       // '->' RIGHT
704       distance.x = -5;
705       if (alt) {
706         // for zoom, we configure the zooming behavior
707         e._in = false;
708         e._fast = false;
709       }
710       
711     } else if (keyCode == 38) {
712       // '^-' TOP
713       distance.y = 5;
714       if (alt) {
715         // for zoom, we configure the zooming behavior
716         e._in = true;
717         e._fast = true;
718       }
719       
720     } else if (keyCode == 40) {
721       // '-v' BOTTOM
722       distance.y = -5;
723       if (alt) {
724         // for zoom, we configure the zooming behavior
725         e._in = false;
726         e._fast = true;
727       }
728       
729     }
730     
731     // attach the distance vector
732     e._distance = distance;
733     
734     // .. fire the event
735     this.dispatchEvent(e);
736     
737   }
738   
739 };
