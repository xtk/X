 45 /**
 46  * Create a displayable object. Objects may have points, colors, a texture, or

 49  * @constructor

226 X.object.prototype.toCSG = function() {
227 
228   var numberOfPoints = this._points.count();
229   
230   var polygons = [];
231   
232   var p = 0;
233   for (p = 0; p < numberOfPoints; p = p + 3) {
234     
235     var point1 = this._points.get(p);
236     var point2 = this._points.get(p + 1);
237     var point3 = this._points.get(p + 2);
238     
239     var normal1 = this._normals.get(p);
240     var normal2 = this._normals.get(p + 1);
241     var normal3 = this._normals.get(p + 2);
242     
243     // get the object color
244     var color = this._color;
245     
246     // if point colors are defined on this X.object, use'em
247     if ((this._colors.length() > 0)) {
248       
249       // we only grab the color of the first point since CSG only supports
250       // colors per triangle not per point
251       color = this._colors.get(p);
252       
253     }
254     
255     //
256     // create a new CSG.Polygon
257     //
258     
259     var vertices = [];
260     vertices.push(new CSG.Vertex(point1, normal1));
261     vertices.push(new CSG.Vertex(point2, normal2));
262     vertices.push(new CSG.Vertex(point3, normal3));
263     
264     polygons.push(new CSG.Polygon(vertices, color));
265     
266   }
267   
268   //
269   // create and return a new CSG object
270   return CSG.fromPolygons(polygons);
271   
272 };

275 X.object.prototype.fromCSG = function(csg) {
276 
277   if (!goog.isDefAndNotNull(csg) || !(csg instanceof CSG)) {
278     
279     throw new X.exception('Invalid CSG object.');
280     
281   }
282   
283   // remove all previous points
284   this._points.clear();
285   this._normals.clear();
286   this._colors.clear();
287   
288   var indexer = new X.indexer();
289   
290   // .. a temp. array to store the triangles using vertex indices
291   var triangles = new Array();
292   
293   // grab points, normals and colors
294   csg.toPolygons().map(function(p) {
295 
296     var indices = p.vertices.map(function(vertex) {
297 
298       vertex.color = p.shared;
299       return indexer.add(vertex);
300     });
301     
302     var i = 2;
303     for (i = 2; i < indices.length; i++) {
304       triangles.push([indices[0], indices[i - 1], indices[i]]);
305     }
306     
307   }.bind(this));
308   
309   // re-map the vertices, normals and colors
310   this.__vertices = indexer.unique().map(function(v) {
311 
312     return [v.pos.x, v.pos.y, v.pos.z];
313   });
314   this.__normals = indexer.unique().map(function(v) {
315 
316     return [v.normal.x, v.normal.y, v.normal.z];
317   });
318   this.__colors = indexer.unique().map(function(v) {
319 
320     if (!v.color) {
321       
322       return null;
323       
324     }
325     return [v.color[0], v.color[1], v.color[2]];
326   });
327   
328   //
329   // setup the points, normals and colors for this X.object
330   // by converting the triangles to the X.object API
331   triangles.map(function(i) {
332 
333     // grab the three vertices of this triangle
334     var i0 = i[0];
335     var i1 = i[1];
336     var i2 = i[2];
337     
338     var vertices = this.__vertices;
339     var normals = this.__normals;
340     var colors = this.__colors;
341     
342     // add the points
343     this._points.add(vertices[i0][0], vertices[i0][1], vertices[i0][2]);
344     this._points.add(vertices[i1][0], vertices[i1][1], vertices[i1][2]);
345     this._points.add(vertices[i2][0], vertices[i2][1], vertices[i2][2]);
346     
347     // add the normals
348     this._normals.add(normals[i0][0], normals[i0][1], normals[i0][2]);
349     this._normals.add(normals[i1][0], normals[i1][1], normals[i1][2]);
350     this._normals.add(normals[i2][0], normals[i2][1], normals[i2][2]);
351     
352     // if colors are set for this triangle, add'em
353     if (colors[i0]) {
354       this._colors.add(colors[i0][0], colors[i0][1], colors[i0][2]);
355     }
356     if (colors[i1]) {
357       this._colors.add(colors[i1][0], colors[i1][1], colors[i1][2]);
358     }
359     if (colors[i2]) {
360       this._colors.add(colors[i2][0], colors[i2][1], colors[i2][2]);
361     }
362     
363   }.bind(this));
364   
365   // we only support CSG in TRIANGLES rendering mode
366   this.setType(X.object.types.TRIANGLES);
367   
368 };

468 /**
469  * Set the object texture. If null is passed, the object will have no texture.
470  * 
471  * @param {?X.texture|string} texture The new texture.
472  * @throws {X.exception} An exception if the given texture is invalid.
473  */
474 X.object.prototype.setTexture = function(texture) {
475 
476   if (!goog.isDefAndNotNull(texture)) {
477     
478     // null textures are allowed
479     this._texture = null;
480     return;
481     
482   }
483   
484   if (goog.isString(texture)) {
485     
486     // a string has to be converted to a new X.texture
487     texture = new X.texture(texture);
488     
489   }
490   
491   if (!(texture instanceof X.texture)) {
492     
493     throw new X.exception('Invalid texture.');
494     
495   }
496   
497   this._texture = texture;
498   
499 };

514 /**
515  * Set the object color. This overrides any point colors.
516  * 
517  * @param {!number} r The Red value in the range of 0..1
518  * @param {!number} g The Green value in the range of 0..1
519  * @param {!number} b The Blue value in the range of 0..1
520  * @throws {X.exception} An exception if the given color values are invalid.
521  */
522 X.object.prototype.setColor = function(r, g, b) {
523 
524   // we accept only numbers as arguments
525   if ((!goog.isNumber(r) && r < 0.0 && r > 1.0) ||
526       (!goog.isNumber(g) && g < 0.0 && g > 1.0) ||
527       (!goog.isNumber(b) && b < 0.0 && b > 1.0)) {
528     
529     throw new X.exception('Invalid color.');
530     
531   }
532   
533   if (this.hasChildren()) {
534     
535     // loop through the children and propagate the new color
536     var children = this.children();
537     var numberOfChildren = children.length;
538     var c = 0;
539     
540     for (c = 0; c < numberOfChildren; c++) {
541       
542       children[c].setColor(r, g, b);
543       
544     }
545     
546   }
547   
548   this._color[0] = r;
549   this._color[1] = g;
550   this._color[2] = b;
551   
552   this._dirty = true;
553   
554 };

557 X.object.prototype.union = function(object) {
558 
559   if (!goog.isDefAndNotNull(object) ||
560       (!(object instanceof CSG) && !(object instanceof X.object))) {
561     
562     throw new X.exception('Invalid object.');
563     
564   }
565   
566   var csg = object;
567   
568   // if we operate on another X.object, we gotta convert
569   if (object instanceof X.object) {
570     
571     csg = csg.toCSG();
572     
573   }
574   
575   var result = new X.object();
576   result.fromCSG(this.toCSG().union(csg));
577   
578   return result;
579   
580 };

711     var numberOfChildren = children.length;

719     
720   }
721   
722   this._visible = visible;

766  *          associated file.

769

771     
772     // if path is null, we reset the associated X.file object
773

779   if (!this._file) {
780     
781     this._file = new X.file();
782     
783   }
784   
785   this._file.setPath(filepath);
