 31 /**
 32  * Create a renderer with the given width and height.
 33  * 
 34  * @constructor
 35  * @extends X.base
 36  */
 37 X.renderer = function(container) {
 38 
 39   // check if a container is passed
 40   if (!goog.isDefAndNotNull(container)) {
 41     
 42     throw new X.exception(
 43         'Fatal: An ID to a valid container (<div>..) is required!');
 44     
 45   }
 46   
 47   // check if the passed container is really valid
 48   var _container = goog.dom.getElement(container);
 49   
 50   if (!goog.dom.isElement(_container) || _container.clientWidth == 0 ||
 51       _container.clientHeight == 0) {
 52     
 53     throw new X.exception(
 54         'Fatal: Could not find the given container or it has an undefined size!');
 55     
 56   }
 57   
 58   //
 59   // call the standard constructor of X.base
 60   goog.base(this);
 61   
 62   //
 63   // class attributes
 64   
 65   /**
 66    * @inheritDoc
 67    * @const
 68    */
 69   this._className = 'renderer';
 70   
 71   /**
 72    * The HTML container of this renderer, E.g. a <div>.
 73    * 
 74    * @type {!Element}
 75    * @protected
 76    */
 77   this._container = goog.dom.getElement(_container);
 78   
 79   /**
 80    * The width of this renderer.
 81    * 
 82    * @type {!number}
 83    * @protected
 84    */
 85   this._width = this._container.clientWidth;
 86   
 87   /**
 88    * The height of this renderer.
 89    * 
 90    * @type {!number}
 91    * @protected
 92    */
 93   this._height = this._container.clientHeight;
 94   
 95   /**
 96    * The Canvas of this renderer.
 97    * 
 98    * @type {?Element}
 99    * @protected
100    */
101   this._canvas = null;
102   
103   /**
104    * The WebGL context of this renderer.
105    * 
106    * @type {?Object}
107    * @protected
108    */
109   this._gl = null;
110   
111   /**
112    * The shader pair for this renderer.
113    * 
114    * @type {?X.shaders}
115    * @protected
116    */
117   this._shaders = null;
118   
119   /**
120    * The compiled shader program of this renderer.
121    * 
122    * @type {?Object}
123    * @protected
124    */
125   this._shaderProgram = null;
126   
127   /**
128    * The camera of this renderer.
129    * 
130    * @type {?X.camera}
131    * @protected
132    */
133   this._camera = null;
134   
135   /**
136    * The interactor of this renderer.
137    * 
138    * @type {?X.interactor}
139    * @protected
140    */
141   this._interactor = null;
142   
143   /**
144    * An array containing the displayable objects of this renderer.
145    * 
146    * @type {!Array}
147    * @protected
148    */
149   this._objects = new Array();
150   
151   this._minX = null;
152   this._maxX = null;
153   this._minY = null;
154   this._maxY = null;
155   this._minZ = null;
156   this._maxZ = null;
157   
158   /**
159    * A hash map of shader attribute pointers.
160    * 
161    * @type {!goog.structs.Map}
162    * @protected
163    */
164   this._attributePointers = new goog.structs.Map();
165   
166   /**
167    * A hash map of vertex buffers of this renderer. Each buffer is associated
168    * with a displayable object using its unique id.
169    * 
170    * @type {!goog.structs.Map}
171    * @protected
172    */
173   this._vertexBuffers = new goog.structs.Map();
174   
175   /**
176    * A hash map of normal buffers of this renderer. Each buffer is associated
177    * with a displayable object using its unique id.
178    * 
179    * @type {!goog.structs.Map}
180    * @protected
181    */
182   this._normalBuffers = new goog.structs.Map();
183   
184   /**
185    * A hash map of color buffers of this renderer. Each buffer is associated
186    * with a displayable object using its unique id.
187    * 
188    * @type {!goog.structs.Map}
189    * @protected
190    */
191   this._colorBuffers = new goog.structs.Map();
192   
193   /**
194    * A hash map of texture position buffers of this renderer. Each buffer is
195    * associated with a displayable object using its unique id.
196    * 
197    * @type {!goog.structs.Map}
198    * @protected
199    */
200   this._texturePositionBuffers = new goog.structs.Map();
201   
202   /**
203    * A hash map of different textures assigned to this renderer. The maximum
204    * number of textures is limited to 32 by WebGL.
205    * 
206    * @type {!goog.structs.Map}
207    * @protected
208    */
209   this._textures = new goog.structs.Map();
210   
211   /**
212    * The loader associated with this renderer.
213    * 
214    * @type {?X.loader}
215    * @protected
216    */
217   this._loader = null;
218   
219   /**
220    * Flag to enable or disable the progress bar during loading.
221    * 
222    * @type {bool}
223    * @protected
224    */
225   this._progressBarEnabled = true;
226   
227   /**
228    * The progressBar of this renderer.
229    * 
230    * @type {?Element}
231    * @protected
232    */
233   this._progressBar = null;
234   
235   /**
236    * The progressBar CSS style DOM element of this renderer.
237    * 
238    * @type {?Element}
239    * @protected
240    */
241   this._progressBarStyle = null;
242   
243   /**
244    * The collection of progressBar CSS definitions.
245    * 
246    * @type {!Array}
247    * @protected
248    */
249   this._progressBarCss = [];
250   // configure some styles
251   var css1 = '.progress-bar-horizontal {\n';
252   css1 += '  position: relative;\n';
253   css1 += '  border: 1px solid #949dad;\n';
254   css1 += '  background: white;\n';
255   css1 += '  padding: 1px;\n';

258   css1 += '  width: 100px;\n';
259   css1 += '  height: 5px;\n';
260   css1 += '}';
261   var css2 = '.progress-bar-thumb {\n';
262   css2 += '  position: relative;\n';
263   css2 += '  background: #F62217;\n';
264   css2 += '  overflow: hidden;\n';
265   css2 += '  width: 0%;\n';
266   css2 += '  height: 100%;\n';
267   css2 += '}';
268   var css3 = '.progress-bar-thumb-done {\n';
269   css3 += '  background: #57E964;\n';
270   css3 += '}';
271   // .. and save them for later use
272   this._progressBarCss = [css1, css2, css3];
273   
274   window.console.log('XTK Release 0 -- http://www.goXTK.com');
275 };

378   }

420 /**
421  * Shows the loading progress bar by modifying the DOM tree.
422  */
423 X.renderer.prototype.showProgressBar_ = function() {
424 
425   // only do the following if the progressBar was not turned off
426   if (this._progressBarEnabled) {
427     
428     // create a progress bar here if this is the first render request and the
429     // loader is working
430     if (!this._progressBar) {
431       
432       // enable relative positioning for the main container
433       // this is required to place the progressBar in the center
434       this.container().style.position = 'relative';
435       
436       // add some CSS to the <head>
437       var head = goog.dom.getDocument().getElementsByTagName('head')[0];
438       var style = goog.dom.createDom('style');
439       style.type = 'text/css';
440       style.media = 'screen';
441       var css = goog.dom.createTextNode(this._progressBarCss[0]);
442       var css2 = goog.dom.createTextNode(this._progressBarCss[1]);
443       var css3 = goog.dom.createTextNode(this._progressBarCss[2]);
444       goog.dom.appendChild(head, style);
445       goog.dom.appendChild(style, css);
446       goog.dom.appendChild(style, css2);
447       goog.dom.appendChild(style, css3);
448       
449       var pb = new goog.ui.ProgressBar();
450       pb.render(this.container());
451       
452       // place the progressBar in the center
453       var pbElement = pb.getElement();
454       pbElement.style.position = 'absolute';
455       pbElement.style.top = (this.height() - 5) / 2;
456       pbElement.style.left = (this.width() - 100) / 2;
457       
458       this._progressBar = pb;
459       this._progressBarStyle = style;
460       
461     }
462     
463   }
464   
465 };

468 /**
469  * Hides the loading progress bar.
470  */
471 X.renderer.prototype.hideProgressBar_ = function() {
472 
473   // only do the following if the progressBar was not turned off
474   if (this._progressBarEnabled) {
475     
476     if (this._progressBar) {
477       
478       // hide the original progress bar
479       goog.dom.removeNode(this._progressBar.getElement());
480       
481       // create a 'fake' new progress bar in place of the original one
482       // the new one always shows 100
483       var pb = new goog.ui.ProgressBar();
484       pb.setValue(100);
485       // this rendering was the trick to force the browser to update the
486       // progressbar
487       pb.render(this.container());
488       
489       // place the progressBar in the center
490       var pbElement = pb.getElement();
491       pbElement.style.position = 'absolute';
492       pbElement.style.top = (this.height() - 5) / 2;
493       pbElement.style.left = (this.width() - 100) / 2;
494       
495       // get the actual progress element of the progressBar
496       var pbBar = goog.dom.getFirstElementChild(pb.getElement());
497       
498       // .. change the color to green
499       pbBar.classList.add('progress-bar-thumb-done');
500       
501       this._progressBar = pb;
502       
503       // wait for a short time
504       this._readyCheckTimer2 = goog.Timer.callOnce(function() {
505 
506         this._readyCheckTimer2 = null;
507         
508         // hide the progress bar
509         if (this._progressBarStyle) {
510           goog.dom.removeNode(this._progressBarStyle);
511         }
512         if (this._progressBar) {
513           goog.dom.removeNode(this._progressBar.getElement());
514         }
515         this._progressBar = null;
516         this._progressBarStyle = null;
517         
518         this.render();
519         
520       }.bind(this), 2000);
521       // .. and jump out
522       return;
523       
524     } // if progressBar still exists
525     
526   } // if progressBar is enabled
527   
528 };

530 /**
531  * Resets the view according to the global bounding box of all associated
532  * objects. This does not trigger rendering.
533  */
534 X.renderer.prototype.resetView = function() {
535 
536   var focus = this._camera.focus();
537   var position = this._camera.position();
538   
539   // if the camera was positioned manually, we do not want to update the
540   // position or focus here
541   if (focus.x != 0 || focus.y != 0 || focus.z != 0 || position.x != 0 ||
542       position.y != 0 || position.z != 100) {
543     
544     this._camera.reset();
545     
546     // and jump out
547     return;
548   }
549   
550   if (!goog.isNull(this._minX) && !goog.isNull(this._maxX) &&
551       !goog.isNull(this._minY) && !goog.isNull(this._maxY) &&
552       !goog.isNull(this._minZ) && !goog.isNull(this._maxZ)) {
553     
554     var focusX = (this._minX + this._maxX) / 2;
555     var focusY = (this._minY + this._maxY) / 2;
556     var focusZ = (this._minZ + this._maxZ) / 2;
557     
558     focus = new goog.math.Vec3(focusX, focusY, focusZ);
559     position = new goog.math.Vec3(focusX, focusY, focusZ + 100);
560     
561   }
562   
563   this._camera.setFocus(focus.x, focus.y, focus.z);
564   this._camera.setPosition(position.x, position.y, position.z);
565   
566 };

581 /**
582  * Create the canvas of this renderer inside the configured container and using
583  * attributes like width, height etc. Then, initialize the WebGL context and
584  * attach all necessary objects (e.g. camera, shaders..). Finally, initialize
585  * the event listeners.
586  * 
587  * @throws {X.exception} An exception if there were problems during
588  *           initialization.
589  */
590 X.renderer.prototype.init = function() {
591 
592   // get the canvas
593   var canvas = this.canvas();
594   
595   // width and height can not be set using CSS but via object properties
596   canvas.width = this.width();
597   canvas.height = this.height();
598   
599   //
600   // append it to the container
601   goog.dom.appendChild(this.container(), canvas);
602   
603   // --------------------------------------------------------------------------
604   //
605   // WebGL Viewport initialization
606   //
607   
608   //
609   // Step1: Get Context of canvas
610   //
611   try {
612     
613     var gl = canvas.getContext('experimental-webgl');
614     
615   } catch (e) {
616     
617     // TODO show dialog
618     throw new X.exception('Fatal: Exception while getting GL Context!\n' + e);
619     
620   }
621   
622   //
623   // Step2: Check if we got the context, if not, WebGL is not supported
624   //
625   if (!gl) {
626     
627     throw new X.exception('Fatal: WebGL not supported!');
628     
629   }
630   
631   //
632   // Step3: Configure the context
633   //
634   try {
635     
636     gl.viewport(0, 0, this.width(), this.height());
637     
638     // configure opacity to 0.0 to overwrite the viewport background-color by
639     // the container color
640     gl.clearColor(0.0, 0.0, 0.0, 0.0);
641     
642     // enable transparency
643     gl.enable(gl.BLEND);
644     gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
645     // enable depth testing
646     gl.enable(gl.DEPTH_TEST);
647     // gl.polygonOffset(1.0, 1.0);
648     // .. with perspective rendering
649     gl.depthFunc(gl.LEQUAL);
650     
651     // clear color and depth buffer
652     gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
653     
654   } catch (e) {
655     
656     // this exception indicates if the browser supports WebGL
657     throw new X.exception('Fatal: Exception while accessing GL Context!\n' + e);
658     
659   }
660   
661   //
662   // WebGL Viewport initialization done
663   // --------------------------------------------------------------------------
664   
665   // now since we have a valid gl viewport, we want to configure the interactor
666   // and camera
667   
668   //
669   // create a new interactor
670   var interactor = new X.interactor(canvas);
671   // TODO flags to disable the interactor on X.renderer-base
672   interactor.observeKeyboard();
673   interactor.observeMouseWheel();
674   interactor.observeMouseClicks();
675   interactor.observeMouseMovement();
676   // .. listen to resetViewEvents
677   goog.events.listen(interactor, X.event.events.RESETVIEW,
678       this.resetViewAndRender.bind(this));
679   
680   //
681   // create a new camera
682   // width and height are required to calculate the perspective
683   var camera = new X.camera(this.width(), this.height());
684   // observe the interactor for user interactions (mouse-movements etc.)
685   camera.observe(interactor);
686   // ..listen to render requests from the camera
687   // these get fired after user-interaction and camera re-positioning to re-draw
688   // all objects
689   goog.events.listen(camera, X.event.events.RENDER, this.render_.bind(this));
690   
691   //
692   // attach all created objects as class attributes
693   // should be one of the last things to do here since we use these attributes
694   // to check if the initialization was completed successfully
695   this._canvas = canvas;
696   this._gl = gl;
697   this._camera = camera;
698   this._interactor = interactor;
699   
700   //
701   // add default shaders to this renderer
702   // it is possible to attach other custom shaders after this init call
703   // also, this has to happen after this._canvas, this._gl and this._camera were
704   // attached to this renderer since we check for these
705   var defaultShaders = new X.shaders();
706   this.addShaders(defaultShaders);
707   
708 };

711 /**
712  * Add a pair of shaders to this renderer. The renderer has to be initialized
713  * before adding the shaders.
714  * 
715  * @param {!X.shaders} shaders The X.shaders pair to add to this renderer.
716  */
717 X.renderer.prototype.addShaders = function(shaders) {
718 
719   // check if the renderer is initialized properly
720   if (!goog.isDefAndNotNull(this._canvas) || !goog.isDefAndNotNull(this._gl) ||
721       !goog.isDefAndNotNull(this._camera)) {
722     
723     throw new X.exception('Fatal: Renderer was not initialized properly!');
724     
725   }
726   
727   // check if the given shaders are valid
728   if (!goog.isDefAndNotNull(shaders) || !(shaders instanceof X.shaders)) {
729     
730     throw new X.exception('Fatal: Could not add shaders!');
731     
732   }
733   
734   // call the validate() method of the shader pair
735   // this will cause exceptions if the validation fails..
736   shaders.validate();
737   
738   // compile the fragment and vertex shaders
739   var glFragmentShader = this._gl.createShader(this._gl.FRAGMENT_SHADER);
740   var glVertexShader = this._gl.createShader(this._gl.VERTEX_SHADER);
741   
742   // attach the sources, defined in the shaders pair
743   this._gl.shaderSource(glFragmentShader, shaders.fragment());
744   this._gl.shaderSource(glVertexShader, shaders.vertex());
745   
746   // start compilation
747   this._gl.compileShader(glFragmentShader);
748   this._gl.compileShader(glVertexShader);
749   
750   if (!this._gl.getShaderParameter(glFragmentShader, this._gl.COMPILE_STATUS)) {
751     
752     throw new X.exception('Fatal: Fragement Shader compilation failed!\n' +
753         this._gl.getShaderInfoLog(glFragmentShader));
754     
755   }
756   
757   if (!this._gl.getShaderParameter(glVertexShader, this._gl.COMPILE_STATUS)) {
758     
759     throw new X.exception('Fatal: Vertex Shader compilation failed!\n' +
760         this._gl.getShaderInfoLog(glVertexShader));
761     
762   }
763   
764   // create a shaderProgram, attach the shaders and link'em all together
765   var shaderProgram = this._gl.createProgram();
766   this._gl.attachShader(shaderProgram, glVertexShader);
767   this._gl.attachShader(shaderProgram, glFragmentShader);
768   this._gl.linkProgram(shaderProgram);
769   
770   if (!this._gl.getProgramParameter(shaderProgram, this._gl.LINK_STATUS)) {
771     
772     throw new X.exception('Fatal: Could not create shader program!');
773     
774   }
775   
776   // activate the new shaderProgram
777   this._gl.useProgram(shaderProgram);
778   
779   // store the index of the position, color, opacity etc. attributes
780   
781   // this._attributePointers.set(key, value)
782   
783   this._vertexPositionAttribute = this._gl.getAttribLocation(shaderProgram,
784       shaders.position());
785   this._gl.enableVertexAttribArray(this._vertexPositionAttribute);
786   
787   this._normalPositionAttribute = this._gl.getAttribLocation(shaderProgram,
788       shaders.normal());
789   this._gl.enableVertexAttribArray(this._normalPositionAttribute);
790   
791   this._vertexColorAttribute = this._gl.getAttribLocation(shaderProgram,
792       shaders.color());
793   this._gl.enableVertexAttribArray(this._vertexColorAttribute);
794   
795   this._texturePositionAttribute = this._gl.getAttribLocation(shaderProgram,
796       shaders.texturePos());
797   this._gl.enableVertexAttribArray(this._texturePositionAttribute);
798   
799   // attach the shaderProgram to this renderer
800   this._shaderProgram = shaderProgram;
801   
802   // attach the shaders to this renderer
803   this._shaders = shaders;
804   
805 };

809  * Add a new displayable object to this renderer. The renderer has to be
810  * initialized before doing so. A X.renderer.render() call has to be initiated
811  * to display added objects.
812  * 
813  * @param {!X.object} object The displayable object to add to this renderer.
814  * @throws {X.exception} An exception if something goes wrong.
815  */
816 X.renderer.prototype.add = function(object) {
817 
818   if (!goog.isDefAndNotNull(this._canvas) || !goog.isDefAndNotNull(this._gl) ||
819       !goog.isDefAndNotNull(this._camera)) {
820     
821     throw new X.exception('Fatal: Renderer was not initialized properly!');
822     
823   }
824   
825   if (!goog.isDefAndNotNull(object) || !(object instanceof X.object)) {
826     
827     throw new X.exception('Fatal: Illegal object!');
828     
829   }
830   
831   // listen to modified events of this object
832   goog.events.listen(object, X.event.events.MODIFIED, this.onModified
833       .bind(this));
834   
835   this.update_(object);
836   
837 };

840 /**
841  * Configure a displayable object within this renderer. The object can be a
842  * newly created one or an existing one. A X.renderer.render() call has to be
843  * initiated to display the object.
844  * 
845  * @param {!X.object} object The displayable object to setup within this
846  *          renderer.
847  * @throws {X.exception} An exception if something goes wrong.
848  * @private
849  */
850 X.renderer.prototype.update_ = function(object) {
851 
852   if (!goog.isDefAndNotNull(this._canvas) || !goog.isDefAndNotNull(this._gl) ||
853       !goog.isDefAndNotNull(this._camera)) {
854     
855     throw new X.exception('Fatal: Renderer was not initialized properly!');
856     
857   }
858   
859   if (!goog.isDefAndNotNull(object) || !(object instanceof X.object)) {
860     
861     throw new X.exception('Fatal: Illegal object!');
862     
863   }
864   
865   // check if object already existed..
866   // if (this._objects.contains(object)) {
867   
868   // TODO
869   if (1 == 2) {
870     // .. it did
871     
872     // remove it from the tree
873     this._objects.remove(object);
874     
875     // clear all glBuffers by reading the X.buffer
876     
877 
878     // VERTICES
879     var oldVertexBuffer = this._vertexBuffers.get(object.id());
880     if (goog.isDefAndNotNull(oldVertexBuffer)) {
881       
882       if (this._gl.isBuffer(oldVertexBuffer.glBuffer())) {
883         
884         this._gl.deleteBuffer(oldVertexBuffer.glBuffer());
885         
886       }
887       
888     }
889     
890     // NORMALS
891     var oldNormalBuffer = this._vertexBuffers.get(object.id());
892     if (goog.isDefAndNotNull(oldNormalBuffer)) {
893       
894       if (this._gl.isBuffer(oldNormalBuffer.glBuffer())) {
895         
896         this._gl.deleteBuffer(oldNormalBuffer.glBuffer());
897         
898       }
899       
900     }
901     
902     // COLORS
903     var oldColorBuffer = this._colorBuffers.get(object.id());
904     if (goog.isDefAndNotNull(oldColorBuffer)) {
905       
906       if (this._gl.isBuffer(oldColorBuffer.glBuffer())) {
907         
908         this._gl.deleteBuffer(oldColorBuffer.glBuffer());
909         
910       }
911       
912     }
913     
914     // TEXTURE
915     var oldTexturePositionBuffer = this._texturePositionBuffers
916         .get(object.id());
917     if (goog.isDefAndNotNull(oldTexturePositionBuffer)) {
918       
919       if (this._gl.isBuffer(oldTexturePositionBuffer.glBuffer())) {
920         
921         this._gl.deleteBuffer(oldTexturePositionBuffer.glBuffer());
922         
923       }
924       
925     }
926     
927   }
928   
929   // MULTI OBJECTS
930   //
931   // objects can have N child objects which again can have M child objects and
932   // so on
933   //
934   // check if this object has children
935   if (object.hasChildren()) {
936     
937     // loop through the children and recursively setup the object
938     var children = object.children();
939     var numberOfChildren = children.length;
940     var c = 0;
941     
942     for (c = 0; c < numberOfChildren; c++) {
943       
944       this.update_(children[c]);
945       
946     }
947     
948   }
949   
950 
951   // here we check if additional loading is necessary
952   // this would be the case if
953   // a) the object has an external texture
954   // b) the object is based on an external file (vtk, stl...)
955   // in these cases, we do not directly update the object but activate the
956   // X.loader to get the externals and then let it call the update method
957   if (goog.isDefAndNotNull(object.texture()) && object.texture().dirty()) {
958     // texture associated to this object and it is dirty..
959     
960     // start loading..
961     this.loader().loadTexture(object);
962     
963     return;
964     
965   } else if (goog.isDefAndNotNull(object.file()) && object.dirty()) {
966     // this object is based on an external file and it is dirty..
967     
968     // start loading..
969     this.loader().loadFile(object);
970     
971     return;
972     
973   }
974   
975   //
976   // VERTICES
977   //
978   var glVertexBuffer = this._gl.createBuffer();
979   
980   var points = object.points();
981   
982   // bind and fill with vertices of current object
983   this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glVertexBuffer);
984   
985   this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(points.all()),
986       this._gl.STATIC_DRAW);
987   
988   // create an X.buffer to store the vertices
989   // every vertex consists of 3 items (x,y,z)
990   var vertexBuffer = new X.buffer(glVertexBuffer, points.count(), 3);
991   
992   this.loader().addProgress(0.3);
993   
994   //
995   // BOUNDING BOX
996   //
997   // The global bounding incorporates all individual bounding boxes of the
998   // objects
999   
1000   var transformationMatrix = object.transform().matrix();
1001   
1002   var tMin = transformationMatrix.multiplyByVector(new goog.math.Vec3(points
1003       .minA(), points.minB(), points.minC()));
1004   var tMax = transformationMatrix.multiplyByVector(new goog.math.Vec3(points
1005       .maxA(), points.maxB(), points.maxC()));
1006   
1007   if (goog.isNull(this._minX) || tMin.x < this._minX) {
1008     this._minX = tMin.x;
1009   }
1010   if (goog.isNull(this._maxX) || tMax.x > this._maxX) {
1011     this._maxX = tMax.x;
1012   }
1013   if (goog.isNull(this._minY) || tMin.y < this._minY) {
1014     this._minY = tMin.y;
1015   }
1016   if (goog.isNull(this._maxY) || tMax.y > this._maxY) {
1017     this._maxY = tMax.y;
1018   }
1019   if (goog.isNull(this._minZ) || tMin.z < this._minZ) {
1020     this._minZ = tMin.z;
1021   }
1022   if (goog.isNull(this._maxZ) || tMax.z > this._maxZ) {
1023     this._maxZ = tMax.z;
1024   }
1025   
1026   // reset the view according to the new bounding box
1027   this.resetView();
1028   
1029   //
1030   // NORMALS
1031   //
1032   var glNormalBuffer = this._gl.createBuffer();
1033   
1034   // bind and fill with normals of current object
1035   this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glNormalBuffer);
1036   this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(object.normals()
1037       .all()), this._gl.STATIC_DRAW);
1038   
1039   // create an X.buffer to store the normals
1040   // every normal consists of 3 items (x,y,z)
1041   var normalBuffer = new X.buffer(glNormalBuffer, object.normals().count(), 3);
1042   
1043   // update progress
1044   this.loader().addProgress(0.3);
1045   
1046   //
1047   // COLORS
1048   //
1049   // Objects can have point colors which can be different for each fragment.
1050   // If no point colors are defined, the object has a solid color.
1051   
1052   // check if we have point colors, then we need to setup the glBuffer and the
1053   // X.buffer
1054   var colorBuffer = null;
1055   
1056   if (object.colors().length() > 0) {
1057     
1058     // yes, there are point colors setup
1059     
1060     // check if the point colors are valid, note that we use the length for this
1061     // check which is slightly faster!
1062     if (object.colors().length() != object.points().length()) {
1063       
1064       // mismatch, this can not work
1065       throw new X.exception('Fatal: Mismatch between points and point colors!');
1066       
1067     }
1068     var glColorBuffer = this._gl.createBuffer();
1069     
1070     // bind and fill with colors defined above
1071     this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glColorBuffer);
1072     this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(object.colors()
1073         .all()), this._gl.STATIC_DRAW);
1074     
1075     // create an X.buffer to store the colors
1076     // every color consists of 3 items (r,g,b)
1077     colorBuffer = new X.buffer(glColorBuffer, object.colors().count(), 3);
1078     
1079   }
1080   
1081   this.loader().addProgress(0.3);
1082   
1083   //
1084   // TEXTURE
1085   //
1086   var texturePositionBuffer = null;
1087   if (goog.isDefAndNotNull(object.texture())) {
1088     // texture associated to this object
1089     
1090     // check if we have a valid texture-to-object's-coordinate map
1091     if (!goog.isDefAndNotNull(object.textureCoordinateMap())) {
1092       
1093       var m = 'Fatal: Can not add an object and texture ';
1094       m += 'without valid coordinate mapping! Set the textureCoordinateMap!';
1095       throw new X.exception(m);
1096       
1097     }
1098     
1099     // setup the glTexture, at this point the image for the texture was already
1100     // loaded thanks to X.loader
1101     var texture = this._gl.createTexture();
1102     
1103     // connect the image and the glTexture
1104     texture.image = object.texture().image();
1105     
1106     //
1107     // activate the texture on the WebGL side
1108     this._textures.set(object.texture().file(), texture);
1109     
1110     this._gl.pixelStorei(this._gl.UNPACK_FLIP_Y_WEBGL, false);
1111     
1112     this._gl.bindTexture(this._gl.TEXTURE_2D, texture);
1113     this._gl.texImage2D(this._gl.TEXTURE_2D, 0, this._gl.RGBA, this._gl.RGBA,
1114         this._gl.UNSIGNED_BYTE, texture.image);
1115     
1116     // TODO different filters?
1117     this._gl.texParameteri(this._gl.TEXTURE_2D, this._gl.TEXTURE_MAG_FILTER,
1118         this._gl.LINEAR);
1119     this._gl.texParameteri(this._gl.TEXTURE_2D, this._gl.TEXTURE_MIN_FILTER,
1120         this._gl.LINEAR);
1121     
1122     // release the texture binding to clear things
1123     this._gl.bindTexture(this._gl.TEXTURE_2D, null);
1124     
1125     // create texture buffer
1126     var glTexturePositionBuffer = this._gl.createBuffer();
1127     
1128     // bind and fill with colors defined above
1129     this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glTexturePositionBuffer);
1130     this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(object
1131         .textureCoordinateMap()), this._gl.STATIC_DRAW);
1132     
1133     // create an X.buffer to store the texture-coordinate map
1134     texturePositionBuffer = new X.buffer(glTexturePositionBuffer, object
1135         .textureCoordinateMap().length, 2);
1136     
1137   } // check if object has a texture
1138   
1139   this.loader().addProgress(0.1);
1140   
1141   //
1142   // FINAL STEPS
1143   //
1144   
1145   // add the object to the internal tree which reflects the rendering order
1146   // (based on opacity)
1147   if (!this._objects.push(object)) {
1148     
1149     throw new X.exception('Fatal: Could not add object to this renderer.');
1150     
1151   }
1152   
1153   // add the buffers for the object to the internal hash maps
1154   this._vertexBuffers.set(object.id(), vertexBuffer);
1155   this._normalBuffers.set(object.id(), normalBuffer);
1156   this._colorBuffers.set(object.id(), colorBuffer);
1157   this._texturePositionBuffers.set(object.id(), texturePositionBuffer);
1158 };

1161 /**
1162  * (Re-)render all associated displayable objects of this renderer. This method
1163  * clears the viewport and re-draws everything by looping through the tree of
1164  * objects. The current perspective and view matrices of the associated camera
1165  * are used to setup the three-dimensional space.
1166  */
1167 X.renderer.prototype.render = function() {
1168 
1169   if (!this._canvas || !this._gl || !this._camera) {
1170     
1171     throw new X.exception('Fatal: The renderer was not initialized properly!');
1172     
1173   }
1174   
1175   // READY CHECK
1176   //
1177   // now we check if we are ready to display everything
1178   // - ready means: all textures loaded and setup, all external files loaded and
1179   // setup and all other objects loaded and setup
1180   //
1181   // if we are not ready, we wait..
1182   // if we are ready, we continue with the rendering
1183   
1184   // let's check if render() was called before and the single-shot timer is
1185   // already there
1186   // f.e., if we are in a setInterval-configured render loop, we do not want to
1187   // create multiple single-shot timers
1188   if (goog.isDefAndNotNull(this._readyCheckTimer)) {
1189     
1190     return;
1191     
1192   }
1193   
1194   //
1195   // LOADING..
1196   //
1197   if (!this.loader().completed()) {
1198     
1199     // we are not ready yet.. the loader is still working;
1200     
1201     this.showProgressBar_();
1202     
1203     // let's check again in a short time
1204     this._readyCheckTimer = goog.Timer.callOnce(function() {
1205 
1206       this._readyCheckTimer = null; // destroy the timer
1207       
1208       // try to render now..
1209       // if the loader is ready it will work, else wise another single-shot gets
1210       // configured in 500 ms
1211       this.render();
1212       
1213     }.bind(this), 100); // check again in 500 ms
1214     
1215     return; // .. and jump out
1216     
1217   } else if (this._progressBar) {
1218     
1219     // we are ready! yahoooo!
1220     // this means the X.loader is done..
1221     this.hideProgressBar_();
1222     
1223     // .. we exit here since the hiding takes some time and automatically
1224     // triggers the rendering when done
1225     return;
1226     
1227   }
1228   //
1229   // END OF LOADING
1230   //
1231   
1232   //
1233   // CURTAIN UP! LET THE SHOW BEGIN..
1234   //
1235   
1236   this.render_();
1237   
1238 };

1241 X.renderer.prototype.render_ = function() {
1242 
1243   // clear the canvas
1244   this._gl.clear(this._gl.COLOR_BUFFER_BIT | this._gl.DEPTH_BUFFER_BIT);
1245   
1246   // grab the current perspective from the camera
1247   var perspectiveMatrix = this._camera.perspective();
1248   
1249   // grab the current view from the camera
1250   var viewMatrix = this._camera.glView();
1251   var viewMatrixInverseTransposed = this._camera.glViewInvertedTransposed();
1252   
1253   // propagate perspective, view and normal matrices to the uniforms of
1254   // the shader
1255   var perspectiveUniformLocation = this._gl.getUniformLocation(
1256       this._shaderProgram, this._shaders.perspective());
1257   
1258   // this._gl.uniformMatrix4fv(perspectiveUniformLocation, false,
1259   // new Float32Array(perspectiveMatrix.flatten()));
1260   this._gl.uniformMatrix4fv(perspectiveUniformLocation, false,
1261       perspectiveMatrix);
1262   
1263   var viewUniformLocation = this._gl.getUniformLocation(this._shaderProgram,
1264       this._shaders.view());
1265   
1266   this._gl.uniformMatrix4fv(viewUniformLocation, false, viewMatrix);
1267   
1268   var normalUniformLocation = this._gl.getUniformLocation(this._shaderProgram,
1269       this._shaders.normalUniform());
1270   
1271   this._gl.uniformMatrix4fv(normalUniformLocation, false,
1272       viewMatrixInverseTransposed);
1273   
1274 
1275   //
1276   // loop through all objects and (re-)draw them
1277   var objects = this._objects;// .getValues();
1278   var numberOfObjects = objects.length;
1279   
1280   var i;
1281   for (i = 0; i < numberOfObjects; ++i) {
1282     
1283     var object = objects[i];
1284     
1285     if (object) {
1286       // we have a valid object
1287       
1288       // check visibility
1289       if (!object.visible()) {
1290         
1291         // not visible, continue to the next one..
1292         continue;
1293         
1294       }
1295       
1296       var id = object.id();
1297       
1298 
1299       var vertexBuffer = this._vertexBuffers.get(id);
1300       var normalBuffer = this._normalBuffers.get(id);
1301       var colorBuffer = this._colorBuffers.get(id);
1302       var texturePositionBuffer = this._texturePositionBuffers.get(id);
1303       
1304       // ..bind the glBuffers
1305       
1306       // VERTICES
1307       this._gl.bindBuffer(this._gl.ARRAY_BUFFER, vertexBuffer.glBuffer());
1308       
1309       this._gl.vertexAttribPointer(this._vertexPositionAttribute, vertexBuffer
1310           .itemSize(), this._gl.FLOAT, false, 0, 0);
1311       
1312       // NORMALS
1313       this._gl.bindBuffer(this._gl.ARRAY_BUFFER, normalBuffer.glBuffer());
1314       
1315       this._gl.vertexAttribPointer(this._normalPositionAttribute, normalBuffer
1316           .itemSize(), this._gl.FLOAT, false, 0, 0);
1317       
1318       // COLORS
1319       var useObjectColorUniformLocation = this._gl.getUniformLocation(
1320           this._shaderProgram, this._shaders.useObjectColor());
1321       
1322       if (goog.isDefAndNotNull(colorBuffer)) {
1323         
1324         // point colors are defined for this object
1325         
1326         // de-activate the useObjectColor flag on the shader
1327         this._gl.uniform1i(useObjectColorUniformLocation, false);
1328         
1329         this._gl.bindBuffer(this._gl.ARRAY_BUFFER, colorBuffer.glBuffer());
1330         
1331         this._gl.vertexAttribPointer(this._vertexColorAttribute, colorBuffer
1332             .itemSize(), this._gl.FLOAT, false, 0, 0);
1333         
1334       } else {
1335         
1336         // we have a fixed object color
1337         
1338         // activate the useObjectColor flag on the shader
1339         this._gl.uniform1i(useObjectColorUniformLocation, true);
1340         
1341         var objectColor = object.color();
1342         
1343         var objectColorUniformLocation = this._gl.getUniformLocation(
1344             this._shaderProgram, this._shaders.objectColor());
1345         
1346         this._gl.uniform3f(objectColorUniformLocation,
1347             parseFloat(objectColor[0]), parseFloat(objectColor[1]),
1348             parseFloat(objectColor[2]));
1349         
1350         // we always have to configure the attribute of the point colors
1351         // even if no point colors are in use
1352         this._gl.vertexAttribPointer(this._vertexColorAttribute, vertexBuffer
1353             .itemSize(), this._gl.FLOAT, false, 0, 0);
1354         
1355       }
1356       
1357       // OPACITY
1358       var objectOpacityUniformLocation = this._gl.getUniformLocation(
1359           this._shaderProgram, this._shaders.objectOpacity());
1360       
1361       this._gl.uniform1f(objectOpacityUniformLocation, parseFloat(object
1362           .opacity()));
1363       
1364       // TEXTURE
1365       var useTextureUniformLocation = this._gl.getUniformLocation(
1366           this._shaderProgram, this._shaders.useTexture());
1367       
1368       if (goog.isDefAndNotNull(object.texture()) &&
1369           goog.isDefAndNotNull(texturePositionBuffer)) {
1370         //
1371         // texture associated to this object
1372         //
1373         
1374         // activate the texture flag on the shader
1375         this._gl.uniform1i(useTextureUniformLocation, true);
1376         
1377         // setup the sampler
1378         var textureSamplerUniformLocation = this._gl.getUniformLocation(
1379             this._shaderProgram, this._shaders.textureSampler());
1380         
1381         // bind the texture
1382         this._gl.activeTexture(this._gl.TEXTURE0);
1383         
1384         // grab the texture from the internal hash map using the filename as the
1385         // key
1386         this._gl.bindTexture(this._gl.TEXTURE_2D, this._textures.get(object
1387             .texture().file()));
1388         this._gl.uniform1i(textureSamplerUniformLocation, 0);
1389         
1390         // propagate the current texture-coordinate-map to WebGL
1391         this._gl.bindBuffer(this._gl.ARRAY_BUFFER, texturePositionBuffer
1392             .glBuffer());
1393         
1394         this._gl.vertexAttribPointer(this._texturePositionAttribute,
1395             texturePositionBuffer.itemSize(), this._gl.FLOAT, false, 0, 0);
1396         
1397       } else {
1398         // no texture for this object
1399         this._gl.uniform1i(useTextureUniformLocation, false);
1400         
1401         // we always have to configure the attribute of the texture positions
1402         // even if no textures are in use
1403         this._gl.vertexAttribPointer(this._texturePositionAttribute,
1404             vertexBuffer.itemSize(), this._gl.FLOAT, false, 0, 0);
1405         
1406       }
1407       
1408       // TRANSFORMS
1409       
1410       // propagate transform to the uniform matrices of the shader
1411       var objectTransformUniformLocation = this._gl.getUniformLocation(
1412           this._shaderProgram, this._shaders.objectTransform());
1413       
1414       this._gl.uniformMatrix4fv(objectTransformUniformLocation, false, object
1415           .transform().glMatrix());
1416       
1417       // .. and draw with the object's draw mode
1418       var drawMode = -1;
1419       if (object.type() == X.object.types.TRIANGLES) {
1420         
1421         drawMode = this._gl.TRIANGLES;
1422         
1423       } else if (object.type() == X.object.types.LINES) {
1424         
1425         this._gl.lineWidth(object.lineWidth());
1426         
1427         drawMode = this._gl.LINES;
1428         
1429       } else if (object.type() == X.object.types.TRIANGLE_STRIPS) {
1430         
1431         drawMode = this._gl.TRIANGLE_STRIP;
1432         
1433       } else if (object.type() == X.object.types.POLYGONS) {
1434         
1435         // TODO right now, this is hacked.. we need to use the Van Gogh
1436         // triangulation algorithm or something faster to properly convert
1437         // POLYGONS to TRIANGLES.
1438         if (vertexBuffer.itemCount() % 3 == 0) {
1439           
1440           drawMode = this._gl.TRIANGLES;
1441           
1442         } else {
1443           
1444           drawMode = this._gl.TRIANGLE_FAN;
1445           
1446         }
1447         
1448       }
1449       
1450       // push it to the GPU, baby..
1451       this._gl.drawArrays(drawMode, 0, vertexBuffer.itemCount());
1452       
1453     } else {
1454       
1455       var message = 'Fatal: Could not retrieve object for (re-)drawing!';
1456       throw new X.exception(message);
1457       
1458     }
1459     
1460   } // loop through objects
1461   
1462 };
