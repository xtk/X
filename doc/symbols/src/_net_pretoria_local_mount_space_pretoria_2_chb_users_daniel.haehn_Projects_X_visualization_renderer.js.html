565 /**
566  * Hides the loading progress bar.
567  */
568 X.renderer.prototype.hideProgressBar_ = function() {
569 
570   // only do the following if the progressBar was not turned off
571   if (this['config']['PROGRESSBAR_ENABLED']) {
572     
573     if (this._progressBar && !this._readyCheckTimer2) {
574       
575       // show a green, full progress bar
576       this._progressBar.done();
577       
578       // wait for a short time
579       this._readyCheckTimer2 = goog.Timer.callOnce(function() {
580 
581         this._readyCheckTimer2 = null;
582         
583         if (this._progressBar) {
584           
585           this._progressBar.kill();
586           this._progressBar = null;
587           
588         }
589         
590         this.render();
591         
592       }.bind(this), 700);
593       // .. and jump out
594       return;
595       
596     } // if progressBar still exists
597     
598   } // if progressBar is enabled
599   
600 };

616 /**
617  * Create the canvas of this renderer inside the configured container and using
618  * attributes like width, height etc. Then, initialize the WebGL context and
619  * attach all necessary objects (e.g. camera, shaders..). Finally, initialize
620  * the event listeners.
621  * 
622  * @throws {Error} An exception if there were problems during initialization.
623  */
624 X.renderer.prototype.init = function() {
625 
626   // get the canvas
627   var canvas = this.canvas();
628   
629   // width and height can not be set using CSS but via object properties
630   canvas.width = this.width();
631   canvas.height = this.height();
632   
633   //
634   // append it to the container
635   goog.dom.appendChild(this.container(), canvas);
636   
637   // --------------------------------------------------------------------------
638   //
639   // WebGL Viewport initialization
640   //
641   
642   //
643   // Step1: Get Context of canvas
644   //
645   try {
646     
647     var gl = canvas.getContext('experimental-webgl');
648     
649     if (!gl) {
650       
651       throw new Error();
652       
653     }
654     
655   } catch (e) {
656     
657     // WebGL is not supported with this browser/machine/gpu
658     
659     // TODO replace by nicer popup
660     
661     //    
662     // 
663     // var errorDialog = new goog.ui.Popup(this.container());
664     // // errorDialog.setContent('mmm');
665     // // errorDialog.setTitle('Ooops..');
666     // // errorDialog.setHasTitleCloseButton(false);
667     // // errorDialog.setButtonSet(null);
668     // errorDialog.setVisible(true);
669     
670     var msg = 'WebGL not supported! See http://wiki.goXTK.com/index.php/X:Browsers for instructions..';
671     
672     alert(msg);
673     
674     throw new Error(msg);
675     
676   }
677   
678   //
679   // Step2: Configure the context
680   //
681   try {
682     
683     gl.viewport(0, 0, this.width(), this.height());
684     
685     // configure opacity to 0.0 to overwrite the viewport background-color by
686     // the container color
687     gl.clearColor(0.0, 0.0, 0.0, 0.0);
688     
689     // enable transparency
690     gl.enable(gl.BLEND);
691     gl.blendEquation(gl.FUNC_ADD);
692     gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
693     // gl.blendFuncSeparate(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA, gl.ONE,
694     // gl.ZERO);
695     // gl.blendFunc(gl.DST_COLOR, gl.ZERO);
696     // gl.blendFunc(gl.ONE_MINUS_SRC_ALPHA, gl.SRC_ALPHA);
697     // // enable depth testing
698     gl.enable(gl.DEPTH_TEST);
699     // // gl.polygonOffset(1.0, 1.0);
700     // // .. with perspective rendering
701     gl.depthFunc(gl.LEQUAL);
702     //    
703     
704 
705 
706     // clear color and depth buffer
707     gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
708     
709     if (this['config']['PICKING_ENABLED']) {
710       //
711       // create a frame buffer for the picking functionality
712       //
713       // inspired by JAX https://github.com/sinisterchipmunk/jax/ and
714       // http://dl.dropbox.com/u/5095342/WebGL/webgldemo3.js
715       //
716       // we basically render into an invisible framebuffer and use a unique
717       // object
718       // color to check which object is where (a simulated Z buffer since we can
719       // not directly access the one from WebGL)
720       var pickFrameBuffer = gl.createFramebuffer();
721       var pickRenderBuffer = gl.createRenderbuffer();
722       var pickTexture = gl.createTexture();
723       
724       gl.bindTexture(gl.TEXTURE_2D, pickTexture);
725       
726       gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGB, this.width(), this.height(), 0,
727           gl.RGB, gl.UNSIGNED_BYTE, null);
728       
729       gl.bindFramebuffer(gl.FRAMEBUFFER, pickFrameBuffer);
730       gl.bindRenderbuffer(gl.RENDERBUFFER, pickRenderBuffer);
731       gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, this
732           .width(), this.height());
733       gl.bindRenderbuffer(gl.RENDERBUFFER, null);
734       
735       gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0,
736           gl.TEXTURE_2D, pickTexture, 0);
737       gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT,
738           gl.RENDERBUFFER, pickRenderBuffer);
739       gl.bindFramebuffer(gl.FRAMEBUFFER, null);
740       
741       this._pickFrameBuffer = pickFrameBuffer;
742       
743     }
744     
745   } catch (e) {
746     
747     // this exception indicates if the browser supports WebGL
748     throw new Error('Exception while accessing GL Context!\n' + e);
749     
750   }
751   
752   //
753   // WebGL Viewport initialization done
754   // --------------------------------------------------------------------------
755   
756   // now since we have a valid gl viewport, we want to configure the interactor
757   // and camera
758   
759   //
760   // create a new interactor
761   var interactor = new X.interactor(canvas);
762   interactor.init();
763   // .. listen to resetViewEvents
764   goog.events.listen(interactor, X.event.events.RESETVIEW,
765       this.resetViewAndRender.bind(this));
766   // .. listen to hoverEvents
767   goog.events.listen(interactor, X.event.events.HOVER, this.onHover.bind(this));
768   
769 
770   //
771   // create a new camera
772   // width and height are required to calculate the perspective
773   var camera = new X.camera(this.width(), this.height());
774   // observe the interactor for user interactions (mouse-movements etc.)
775   camera.observe(interactor);
776   // ..listen to render requests from the camera
777   // these get fired after user-interaction and camera re-positioning to re-draw
778   // all objects
779   goog.events.listen(camera, X.event.events.RENDER, this.render_.bind(this,
780       false, false));
781   
782   //
783   // attach all created objects as class attributes
784   // should be one of the last things to do here since we use these attributes
785   // to check if the initialization was completed successfully
786   this._canvas = canvas;
787   this._gl = gl;
788   this._camera = camera;
789   this._interactor = interactor;
790   
791   //
792   // add default shaders to this renderer
793   // it is possible to attach other custom shaders after this init call
794   // also, this has to happen after this._canvas, this._gl and this._camera were
795   // attached to this renderer since we check for these
796   var defaultShaders = new X.shaders();
797   this.addShaders(defaultShaders);
798   
799 };

802 /**
803  * Add a pair of shaders to this renderer. The renderer has to be initialized
804  * before adding the shaders.
805  * 
806  * @param {!X.shaders} shaders The X.shaders pair to add to this renderer.
807  */
808 X.renderer.prototype.addShaders = function(shaders) {
809 
810   // check if the renderer is initialized properly
811   if (!goog.isDefAndNotNull(this._canvas) || !goog.isDefAndNotNull(this._gl) ||
812       !goog.isDefAndNotNull(this._camera)) {
813     
814     throw new Error('Renderer was not initialized properly.');
815     
816   }
817   
818   // check if the given shaders are valid
819   if (!goog.isDefAndNotNull(shaders) || !(shaders instanceof X.shaders)) {
820     
821     throw new Error('Could not add shaders.');
822     
823   }
824   
825   // call the validate() method of the shader pair
826   // this will cause an exception if the validation fails..
827   shaders.validate();
828   
829   // compile the fragment and vertex shaders
830   var glFragmentShader = this._gl.createShader(this._gl.FRAGMENT_SHADER);
831   var glVertexShader = this._gl.createShader(this._gl.VERTEX_SHADER);
832   
833   // attach the sources, defined in the shaders pair
834   this._gl.shaderSource(glFragmentShader, shaders.fragment());
835   this._gl.shaderSource(glVertexShader, shaders.vertex());
836   
837   // start compilation
838   this._gl.compileShader(glFragmentShader);
839   this._gl.compileShader(glVertexShader);
840   
841   if (!this._gl.getShaderParameter(glFragmentShader, this._gl.COMPILE_STATUS)) {
842     
843     throw new Error('Fragement Shader compilation failed!\n' +
844         this._gl.getShaderInfoLog(glFragmentShader));
845     
846   }
847   
848   if (!this._gl.getShaderParameter(glVertexShader, this._gl.COMPILE_STATUS)) {
849     
850     throw new Error('Vertex Shader compilation failed!\n' +
851         this._gl.getShaderInfoLog(glVertexShader));
852     
853   }
854   
855   // create a shaderProgram, attach the shaders and link'em all together
856   var shaderProgram = this._gl.createProgram();
857   this._gl.attachShader(shaderProgram, glVertexShader);
858   this._gl.attachShader(shaderProgram, glFragmentShader);
859   this._gl.linkProgram(shaderProgram);
860   
861   if (!this._gl.getProgramParameter(shaderProgram, this._gl.LINK_STATUS)) {
862     
863     throw new Error('Could not create shader program!\n' +
864         this._gl.getShaderInfoLog(glFragmentShader) + '\n' +
865         this._gl.getShaderInfoLog(glVertexShader) + '\n' +
866         this._gl.getProgramInfoLog(shaderProgram));
867     
868   }
869   
870   // activate the new shaderProgram
871   this._gl.useProgram(shaderProgram);
872   
873   // attach the shaderProgram to this renderer
874   this._shaderProgram = shaderProgram;
875   
876   // store the pointers to the shaders' attributes
877   var attributes = Object.keys(X.shaders.attributes);
878   
879   attributes.forEach(function(a) {
880 
881     a = eval("X.shaders.attributes." + a);
882     this._attributePointers.set(a, this._gl.getAttribLocation(
883         this._shaderProgram, a));
884     this._gl.enableVertexAttribArray(this._attributePointers.get(a));
885     
886   }.bind(this));
887   
888   // store the pointers to the shaders' uniforms
889   var uniforms = Object.keys(X.shaders.uniforms);
890   
891   uniforms.forEach(function(u) {
892 
893     u = eval("X.shaders.uniforms." + u);
894     this._uniformLocations.set(u, this._gl.getUniformLocation(
895         this._shaderProgram, u));
896     
897   }.bind(this));
898   
899   // finally, attach the shaders to this renderer
900   this._shaders = shaders;
901   
902 };

924 /**
925  * Configure a displayable object within this renderer. The object can be a
926  * newly created one or an existing one. A X.renderer.render() call has to be
927  * initiated to display the object.
928  * 
929  * @param {!X.object} object The displayable object to setup within this
930  *          renderer.
931  * @throws {Error} An exception if something goes wrong.
932  * @private
933  */
934 X.renderer.prototype.update_ = function(object) {
935 
936   if (!goog.isDefAndNotNull(this._canvas) || !goog.isDefAndNotNull(this._gl) ||
937       !goog.isDefAndNotNull(this._camera)) {
938     
939     throw new Error('Renderer was not initialized properly.');
940     
941   }
942   
943   if (!goog.isDefAndNotNull(object) || !(object instanceof X.object)) {
944     
945     throw new Error('Illegal object.');
946     
947   }
948   
949   // check if object already existed..
950   var existed = false;
951   
952   if (this.get(object.id())) {
953     // this means, we are updating
954     existed = true;
955     
956   }
957   
958   // listen to modified events of this object, if we didn't do that before
959   if (!goog.events.hasListener(object, X.event.events.MODIFIED)) {
960     
961     goog.events.listen(object, X.event.events.MODIFIED, this.onModified
962         .bind(this));
963     
964   }
965   
966   var id = object.id();
967   var points = object.points();
968   var normals = object.normals();
969   var colors = object.colors();
970   var texture = object.texture();
971   var file = object.file();
972   var transform = object.transform();
973   
974   // here we check if additional loading is necessary
975   // this would be the case if
976   // a) the object has an external texture
977   // b) the object is based on an external file (vtk, stl...)
978   // in these cases, we do not directly update the object but activate the
979   // X.loader to get the externals and then let it call the update method
980   if (goog.isDefAndNotNull(texture) && goog.isDefAndNotNull(texture.file()) &&
981       texture.file().dirty()) {
982     // a texture file is associated to this object and it is dirty..
983     
984     // start loading..
985     this.loader().loadTexture(object);
986     
987     return;
988     
989   } else if (goog.isDefAndNotNull(file) && file.dirty()) {
990     // this object is based on an external file and it is dirty..
991     
992     // start loading..
993     this.loader().loadFile(object);
994     
995     return;
996     
997   }
998   
999   // MULTI OBJECTS
1000   //
1001   // objects can have N child objects which again can have M child objects and
1002   // so on
1003   //
1004   // check if this object has children
1005   if (object.dirty() && object.hasChildren()) {
1006     
1007     // loop through the children and recursively setup the object
1008     var children = object.children();
1009     var numberOfChildren = children.length;
1010     var c = 0;
1011     
1012     for (c = 0; c < numberOfChildren; c++) {
1013       
1014       this.update_(children[c]);
1015       
1016     }
1017     
1018   }
1019   
1020   // check if this is an empty object, if yes, jump out
1021   // empty objects can be used to group objects
1022   if (points.count() == 0) {
1023     
1024     object.setClean();
1025     return;
1026     
1027   }
1028   
1029 
1030   // a simple locking mechanism to prevent multiple calls when using
1031   // asynchronous requests
1032   var counter = 0;
1033   while (this._locked) {
1034     
1035     // wait
1036     counter++;
1037     
1038   }
1039   
1040   this._locked = true;
1041   
1042 
1043   //
1044   // BOUNDING BOX
1045   //
1046   // The global bounding incorporates all individual bounding boxes of the
1047   // objects. This bounding box only changes if either the points or the
1048   // transform are dirty.
1049   if (points.dirty() || transform.dirty()) {
1050     var transformationMatrix = transform.matrix();
1051     
1052     var tMin = transformationMatrix.multiplyByVector(new goog.math.Vec3(points
1053         .minA(), points.minB(), points.minC()));
1054     var tMax = transformationMatrix.multiplyByVector(new goog.math.Vec3(points
1055         .maxA(), points.maxB(), points.maxC()));
1056     
1057     if (goog.isNull(this._minX) || tMin.x < this._minX) {
1058       this._minX = tMin.x;
1059     }
1060     if (goog.isNull(this._maxX) || tMax.x > this._maxX) {
1061       this._maxX = tMax.x;
1062     }
1063     if (goog.isNull(this._minY) || tMin.y < this._minY) {
1064       this._minY = tMin.y;
1065     }
1066     if (goog.isNull(this._maxY) || tMax.y > this._maxY) {
1067       this._maxY = tMax.y;
1068     }
1069     if (goog.isNull(this._minZ) || tMin.z < this._minZ) {
1070       this._minZ = tMin.z;
1071     }
1072     if (goog.isNull(this._maxZ) || tMax.z > this._maxZ) {
1073       this._maxZ = tMax.z;
1074     }
1075     // we always keep track of the current center position
1076     this._center = [(this._minX + this._maxX) / 2,
1077                     (this._minY + this._maxY) / 2,
1078                     (this._minZ + this._maxZ) / 2];
1079     
1080     // only set the transform clean since we still need to look at the points
1081     transform.setClean();
1082   }
1083   
1084 
1085   //
1086   // VERTICES
1087   //
1088   
1089   if (existed && points.dirty()) {
1090     
1091     // this means the object already existed and the points are dirty
1092     // therefore, we delete the old gl buffers
1093     
1094     // remove old vertex buffer
1095     var oldVertexBuffer = this._vertexBuffers.get(id);
1096     if (goog.isDefAndNotNull(oldVertexBuffer)) {
1097       
1098       if (this._gl.isBuffer(oldVertexBuffer._glBuffer)) {
1099         
1100         this._gl.deleteBuffer(oldVertexBuffer._glBuffer);
1101         
1102       }
1103       
1104     }
1105     
1106   }
1107   
1108   var vertexBuffer = null;
1109   
1110   if (!existed || points.dirty()) {
1111     
1112     // the object either did not exist or the points are dirty, so we re-create
1113     // the gl buffers and reset the bounding box
1114     
1115     var glVertexBuffer = this._gl.createBuffer();
1116     
1117     // bind and fill with vertices of current object
1118     this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glVertexBuffer);
1119     
1120     this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(points.all()),
1121         this._gl.STATIC_DRAW);
1122     
1123     // create an X.buffer to store the vertices
1124     // every vertex consists of 3 items (x,y,z)
1125     vertexBuffer = new X.buffer(glVertexBuffer, points.count(), 3);
1126     
1127     points.setClean();
1128     
1129   } else {
1130     
1131     // the points are not dirty and the object already existed, so use the old
1132     // buffer
1133     vertexBuffer = this._vertexBuffers.get(id);
1134     
1135   }
1136   
1137   this.loader().addProgress(0.3);
1138   
1139 
1140   //
1141   // NORMALS
1142   //
1143   
1144   if (existed && normals.dirty()) {
1145     
1146     // this means the object already existed and the points are dirty
1147     // therefore, we delete the old gl buffers
1148     
1149     // remove old normals buffer
1150     var oldNormalBuffer = this._vertexBuffers.get(id);
1151     if (goog.isDefAndNotNull(oldNormalBuffer)) {
1152       
1153       if (this._gl.isBuffer(oldNormalBuffer._glBuffer)) {
1154         
1155         this._gl.deleteBuffer(oldNormalBuffer._glBuffer);
1156         
1157       }
1158       
1159     }
1160     
1161   }
1162   
1163   var normalBuffer = null;
1164   
1165   if (!existed || normals.dirty()) {
1166     
1167     // the object either did not exist or the normals are dirty, so we re-create
1168     // the gl buffers
1169     
1170     var glNormalBuffer = this._gl.createBuffer();
1171     
1172     // bind and fill with normals of current object
1173     this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glNormalBuffer);
1174     this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(normals.all()),
1175         this._gl.STATIC_DRAW);
1176     
1177     // create an X.buffer to store the normals
1178     // every normal consists of 3 items (x,y,z)
1179     normalBuffer = new X.buffer(glNormalBuffer, normals.count(), 3);
1180     
1181     normals.setClean();
1182     
1183   } else {
1184     
1185     // the normals are not dirty and the object already existed, so use the old
1186     // buffer
1187     normalBuffer = this._normalBuffers.get(id);
1188     
1189   }
1190   
1191   // update progress
1192   this.loader().addProgress(0.3);
1193   
1194 
1195   //
1196   // COLORS
1197   //
1198   // Objects can have point colors which can be different for each fragment.
1199   // If no point colors are defined, the object has a solid color.
1200   
1201   if (existed && colors.dirty()) {
1202     
1203     // this means the object already existed and the colors are dirty
1204     // therefore, we delete the old gl buffers
1205     
1206     var oldColorBuffer = this._colorBuffers.get(id);
1207     if (goog.isDefAndNotNull(oldColorBuffer)) {
1208       
1209       if (this._gl.isBuffer(oldColorBuffer._glBuffer)) {
1210         
1211         this._gl.deleteBuffer(oldColorBuffer._glBuffer);
1212         
1213       }
1214       
1215     }
1216   }
1217   
1218   // check if we have point colors, then we need to setup the glBuffer and the
1219   // X.buffer
1220   var colorBuffer = null;
1221   
1222   if (colors.length() > 0) {
1223     
1224     // yes, there are point colors setup
1225     
1226     if (!existed || colors.dirty()) {
1227       
1228       // the object either did not exist or the colors are dirty, so we
1229       // re-create the gl buffers
1230       
1231       // check if the point colors are valid, note that we use the length for
1232       // this
1233       // check which is slightly faster!
1234       if (colors.length() != points.length()) {
1235         
1236         // mismatch, this can not work
1237         throw new Error('Mismatch between points and point colors.');
1238         
1239       }
1240       var glColorBuffer = this._gl.createBuffer();
1241       
1242       // bind and fill with colors defined above
1243       this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glColorBuffer);
1244       this._gl.bufferData(this._gl.ARRAY_BUFFER,
1245           new Float32Array(colors.all()), this._gl.STATIC_DRAW);
1246       
1247       // create an X.buffer to store the colors
1248       // every color consists of 3 items (r,g,b)
1249       colorBuffer = new X.buffer(glColorBuffer, colors.count(), 3);
1250       
1251       colors.setClean();
1252       
1253     } else {
1254       
1255       // the colors are not dirty and the object already existed, so use the old
1256       // buffer
1257       colorBuffer = this._colorBuffers.get(id);
1258       
1259     }
1260     
1261   }
1262   
1263   this.loader().addProgress(0.3);
1264   
1265   //
1266   // TEXTURE
1267   //
1268   
1269   if (existed && goog.isDefAndNotNull(texture) && texture.dirty()) {
1270     
1271     // this means the object already existed and the texture is dirty
1272     // therefore, we delete the old gl buffers
1273     
1274     var oldTexturePositionBuffer = this._texturePositionBuffers.get(id);
1275     if (goog.isDefAndNotNull(oldTexturePositionBuffer)) {
1276       
1277       if (this._gl.isBuffer(oldTexturePositionBuffer._glBuffer)) {
1278         
1279         this._gl.deleteBuffer(oldTexturePositionBuffer._glBuffer);
1280         
1281       }
1282       
1283     }
1284   }
1285   
1286   var texturePositionBuffer = null;
1287   if (goog.isDefAndNotNull(texture)) {
1288     // texture associated to this object
1289     
1290     if (!existed || texture.dirty()) {
1291       
1292       // the object either did not exist or the texture is dirty, so we
1293       // re-create the gl buffers
1294       
1295       var textureCoordinateMap = object.textureCoordinateMap();
1296       
1297       // check if we have a valid texture-to-object's-coordinate map
1298       if (!goog.isDefAndNotNull(textureCoordinateMap)) {
1299         
1300         var m = 'Can not add an object and texture ';
1301         m += 'without valid coordinate mapping! Set the textureCoordinateMap!';
1302         throw new Error(m);
1303         
1304       }
1305       
1306       // setup the glTexture, at this point the image for the texture was
1307       // already
1308       // loaded thanks to X.loader
1309       var glTexture = this._gl.createTexture();
1310       
1311       // connect the image and the glTexture
1312       glTexture.image = texture.image();
1313       
1314       //
1315       // activate the texture on the WebGL side
1316       this._textures.set(texture.id(), glTexture);
1317       
1318       this._gl.bindTexture(this._gl.TEXTURE_2D, glTexture);
1319       if (texture.rawData()) {
1320         
1321         // use rawData rather than loading an imagefile
1322         this._gl.texImage2D(this._gl.TEXTURE_2D, 0, this._gl.RGBA, texture
1323             .rawDataWidth(), texture.rawDataHeight(), 0, this._gl.RGBA,
1324             this._gl.UNSIGNED_BYTE, texture.rawData());
1325         
1326         this._gl.texParameteri(this._gl.TEXTURE_2D, this._gl.TEXTURE_WRAP_S,
1327             this._gl.CLAMP_TO_EDGE);
1328         this._gl.texParameteri(this._gl.TEXTURE_2D, this._gl.TEXTURE_WRAP_T,
1329             this._gl.CLAMP_TO_EDGE);
1330         
1331         // we do not want to flip here
1332         this._gl.pixelStorei(this._gl.UNPACK_FLIP_Y_WEBGL, true);
1333         
1334       } else {
1335         
1336         // use an imageFile for the texture
1337         this._gl.texImage2D(this._gl.TEXTURE_2D, 0, this._gl.RGBA,
1338             this._gl.RGBA, this._gl.UNSIGNED_BYTE, glTexture.image);
1339         
1340         this._gl.pixelStorei(this._gl.UNPACK_FLIP_Y_WEBGL, false);
1341         
1342       }
1343       
1344       // TODO different filters?
1345       this._gl.texParameteri(this._gl.TEXTURE_2D, this._gl.TEXTURE_MAG_FILTER,
1346           this._gl.LINEAR);
1347       this._gl.texParameteri(this._gl.TEXTURE_2D, this._gl.TEXTURE_MIN_FILTER,
1348           this._gl.LINEAR);
1349       
1350       // release the texture binding to clear things
1351       this._gl.bindTexture(this._gl.TEXTURE_2D, null);
1352       
1353       // create texture buffer
1354       var glTexturePositionBuffer = this._gl.createBuffer();
1355       
1356       // bind and fill with colors defined above
1357       this._gl.bindBuffer(this._gl.ARRAY_BUFFER, glTexturePositionBuffer);
1358       this._gl.bufferData(this._gl.ARRAY_BUFFER, new Float32Array(
1359           textureCoordinateMap), this._gl.STATIC_DRAW);
1360       
1361       // create an X.buffer to store the texture-coordinate map
1362       texturePositionBuffer = new X.buffer(glTexturePositionBuffer,
1363           textureCoordinateMap.length, 2);
1364       
1365       texture.setClean();
1366       
1367     } else {
1368       
1369       // the texture is not dirty and the object already existed, so use the old
1370       // buffer
1371       texturePositionBuffer = this._texturePositionBuffers.get(id);
1372       
1373     }
1374     
1375     // dirty check
1376     
1377   } // check if object has a texture
1378   
1379   this.loader().addProgress(0.1);
1380   
1381   //
1382   // FINAL STEPS
1383   //
1384   
1385   // add the object to the internal tree which reflects the rendering order
1386   // (based on opacity)
1387   if (!existed) {
1388     if (!this._objects.add(object)) {
1389       
1390       throw new Error('Could not add object to this renderer.');
1391       
1392     }
1393   }
1394   
1395   // add the buffers for the object to the internal hash maps
1396   // at this point the buffers are: either null (if possible), a newly generated
1397   // one or an old one
1398   this._vertexBuffers.set(id, vertexBuffer);
1399   this._normalBuffers.set(id, normalBuffer);
1400   this._colorBuffers.set(id, colorBuffer);
1401   this._texturePositionBuffers.set(id, texturePositionBuffer);
1402   
1403   // clean the object
1404   object.setClean();
1405   
1406   // unlock
1407   this._locked = false;
1408   
1409 };

1412 /**
1413  * (Re-)render all associated displayable objects of this renderer. This method
1414  * clears the viewport and re-draws everything by looping through the tree of
1415  * objects. The current perspective and view matrices of the associated camera
1416  * are used to setup the three-dimensional space.
1417  */
1418 X.renderer.prototype.render = function() {
1419 
1420   if (!this._canvas || !this._gl || !this._camera) {
1421     
1422     throw new Error('The renderer was not initialized properly.');
1423     
1424   }
1425   
1426   // READY CHECK
1427   //
1428   // now we check if we are ready to display everything
1429   // - ready means: all textures loaded and setup, all external files loaded and
1430   // setup and all other objects loaded and setup
1431   //
1432   // if we are not ready, we wait..
1433   // if we are ready, we continue with the rendering
1434   
1435   // let's check if render() was called before and the single-shot timer is
1436   // already there
1437   // f.e., if we are in a setInterval-configured render loop, we do not want to
1438   // create multiple single-shot timers
1439   if (goog.isDefAndNotNull(this._readyCheckTimer)) {
1440     
1441     return;
1442     
1443   }
1444   
1445   //
1446   // LOADING..
1447   //
1448   if (!this.loader().completed()) {
1449     
1450     // we are not ready yet.. the loader is still working;
1451     
1452     this.showProgressBar_();
1453     
1454     // let's check again in a short time
1455     this._readyCheckTimer = goog.Timer.callOnce(function() {
1456 
1457       this._readyCheckTimer = null; // destroy the timer
1458       
1459       // try to render now..
1460       // if the loader is ready it will work, else wise another single-shot gets
1461       // configured in 500 ms
1462       this.render();
1463       
1464     }.bind(this), 100); // check again in 500 ms
1465     
1466     return; // .. and jump out
1467     
1468   } else if (this._progressBar) {
1469     
1470     // we are ready! yahoooo!
1471     // this means the X.loader is done..
1472     this.hideProgressBar_();
1473     
1474     // call the onShowtime function which can be overloaded
1475     eval("this.onShowtime()");
1476     
1477     // .. we exit here since the hiding takes some time and automatically
1478     // triggers the rendering when done
1479     return;
1480     
1481   }
1482   //
1483   // END OF LOADING
1484   //
1485   
1486   //
1487   // CURTAIN UP! LET THE SHOW BEGIN..
1488   //
1489   this.render_(false, true);
1490   
1491 };

1655 /**
1656  * Calculates the distance for each associated X.object and orders the AVL tree
1657  * accordingly from back-to-front while fully opaque objects are drawn first.
1658  * Jumps out as early as possible if all objects are fully opaque.
1659  */
1660 X.renderer.prototype.order_ = function() {
1661 
1662   // by default we do not want to update the tree
1663   var reSortTreeRequired = false;
1664   
1665   var objects = this._objects.getValues();
1666   var numberOfObjects = objects.length;
1667   
1668   var i;
1669   i = numberOfObjects - 1;
1670   do {
1671     
1672     var object = objects[i];
1673     if (!object) {
1674       return;
1675     }
1676     
1677     var opacity = object['_opacity'];
1678     
1679     // if we order X.slice-s of an X.volume, we want to use the opacity of the
1680     // parent volume but only in volume rendering mode
1681     var volume = object._volume;
1682     if (object instanceof X.slice) {
1683       
1684       if (volume && volume['_volumeRendering']) {
1685         
1686         // X.slices are only transparent in volume rendering mode
1687         opacity = object['_opacity'] = object._volume['_opacity'];
1688         
1689       } else {
1690         
1691         // no volume rendering, so the X.slice is fully opaque
1692         opacity = object['_opacity'] = 1.0;
1693         
1694       }
1695       
1696     }
1697     
1698     if (opacity == 1) {
1699       
1700       // this object is fully opaque, we do not need to calculate the distance
1701       continue;
1702       
1703     }
1704     
1705     var centroid = object._points._centroid;
1706     var centroidVector = new goog.math.Vec3(centroid[0], centroid[1],
1707         centroid[2]);
1708     var transformedCentroidVector = object._transform._matrix
1709         .multiplyByVector(centroidVector);
1710     var realCentroidVector = this._camera._view
1711         .multiplyByVector(transformedCentroidVector);
1712     var distanceFromEye = goog.math.Vec3.distance(this._camera._focus,
1713         realCentroidVector);
1714     objects[i].distance = distanceFromEye;
1715     
1716     // we need to update the tree
1717     reSortTreeRequired = true;
1718     
1719   } while (i--);
1720   
1721   // only re-sort the tree if required
1722   if (reSortTreeRequired) {
1723     
1724     // re-sort the tree by clearing and re-adding all objects
1725     // should be fast..
1726     this._objects.clear();
1727     
1728     i = numberOfObjects - 1;
1729     do {
1730       
1731       this._objects.add(objects[i]);
1732       
1733     } while (i--);
1734     
1735   }
1736   
1737 };

1778 /**
1779  * Internal function to perform the actual rendering by looping through all
1780  * associated X.objects.
1781  * 
1782  * @param {boolean} picking If TRUE, render to a framebuffer to perform picking -
1783  *          if FALSE render to the canvas viewport.
1784  * @param {?boolean=} invoked If TRUE, the render counts as invoked and f.e.
1785  *          statistics are generated.
1786  * @private
1787  */
1788 X.renderer.prototype.render_ = function(picking, invoked) {
1789 
1790   // picking = false;
1791   // for ( var y = 0; y < this._topLevelObjects.length; y++) {
1792   //    
1793   // var topLevelObject = this._topLevelObjects[y];
1794   //    
1795   // if (topLevelObject.hasChildren()) {
1796   //      
1797   // // this.generateTree_(topLevelObject, 0);
1798   //      
1799   // }
1800   //    
1801   // }
1802   
1803   if (picking) {
1804     
1805     // we are in picking mode, so use the framebuffer rather than the canvas
1806     this._gl.bindFramebuffer(this._gl.FRAMEBUFFER, this._pickFrameBuffer);
1807     
1808   } else {
1809     
1810     // disable the framebuffer
1811     this._gl.bindFramebuffer(this._gl.FRAMEBUFFER, null);
1812     
1813   }
1814   
1815   // clear the canvas
1816   this._gl.clear(this._gl.COLOR_BUFFER_BIT | this._gl.DEPTH_BUFFER_BIT);
1817   
1818   // grab the current perspective from the camera
1819   var perspectiveMatrix = this._camera._perspective;
1820   
1821   // grab the current view from the camera
1822   var viewMatrix = this._camera._glView;
1823   
1824   // propagate perspective and view matrices to the uniforms of
1825   // the shader
1826   this._gl.uniformMatrix4fv(this._uniformLocations
1827       .get(X.shaders.uniforms.PERSPECTIVE), false, perspectiveMatrix);
1828   
1829   this._gl.uniformMatrix4fv(
1830       this._uniformLocations.get(X.shaders.uniforms.VIEW), false, viewMatrix);
1831   
1832   // propagate the objects' center to the shader
1833   //
1834   var center = this._center;
1835   this._gl.uniform3f(this._uniformLocations.get(X.shaders.uniforms.CENTER),
1836       parseFloat(center[0]), parseFloat(center[1]), parseFloat(center[2]));
1837   
1838   //
1839   // re-order the objects, but only if enabled.
1840   // this ordering should be disabled if the objects' opacity settings are not
1841   // used or if a large number of objects are associated
1842   if (this['config']['ORDERING_ENABLED']) {
1843     
1844     this.order_();
1845     
1846   }
1847   
1848   var statisticsEnabled = (!picking && goog.isDefAndNotNull(invoked) && invoked && this['config']['STATISTICS_ENABLED']);
1849   if (statisticsEnabled) {
1850     
1851     // for statistics
1852     var verticesCounter = 0;
1853     var trianglesCounter = 0;
1854     var linesCounter = 0;
1855     var pointsCounter = 0;
1856     
1857   }
1858   
1859   //
1860   // orient volumes for proper volume rendering - if there are any,
1861   // this means, depending on the direction of the eye, we use the slice stack
1862   // of a specific axis to create the tiled texture
1863   var i;
1864   var topLevelObjectsLength = this._topLevelObjects.length;
1865   for (i = 0; i < topLevelObjectsLength; ++i) {
1866     var topLevelObject = this._topLevelObjects[i];
1867     if (topLevelObject instanceof X.volume) {
1868       this.orientVolume_(topLevelObject);
1869     }
1870   }
1871   
1872   //
1873   // caching for multiple objects
1874   //
1875   var aPointers = this._attributePointers;
1876   var aPosition = aPointers.get(X.shaders.attributes.VERTEXPOSITION);
1877   var aNormal = aPointers.get(X.shaders.attributes.VERTEXNORMAL);
1878   var aColor = aPointers.get(X.shaders.attributes.VERTEXCOLOR);
1879   var aTexturePosition = aPointers.get(X.shaders.attributes.VERTEXTEXTUREPOS);
1880   
1881   var uLocations = this._uniformLocations;
1882   var uUsePicking = uLocations.get(X.shaders.uniforms.USEPICKING);
1883   var uUseObjectColor = uLocations.get(X.shaders.uniforms.USEOBJECTCOLOR);
1884   var uObjectColor = uLocations.get(X.shaders.uniforms.OBJECTCOLOR);
1885   var uObjectOpacity = uLocations.get(X.shaders.uniforms.OBJECTOPACITY);
1886   var uUseTexture = uLocations.get(X.shaders.uniforms.USETEXTURE);
1887   var uTextureSampler = uLocations.get(X.shaders.uniforms.TEXTURESAMPLER);
1888   var uVolumeLowerThreshold = uLocations
1889       .get(X.shaders.uniforms.VOLUMELOWERTHRESHOLD);
1890   var uVolumeUpperThreshold = uLocations
1891       .get(X.shaders.uniforms.VOLUMEUPPERTHRESHOLD);
1892   var uVolumeScalarMin = uLocations.get(X.shaders.uniforms.VOLUMESCALARMIN);
1893   var uVolumeScalarMax = uLocations.get(X.shaders.uniforms.VOLUMESCALARMAX);
1894   var uObjectTransform = uLocations.get(X.shaders.uniforms.OBJECTTRANSFORM);
1895   var uPointSize = uLocations.get(X.shaders.uniforms.POINTSIZE);
1896   
1897   //
1898   // loop through all objects and (re-)draw them
1899   var objects = this._objects.getValues();
1900   var numberOfObjects = objects.length;
1901   i = numberOfObjects;
1902   do {
1903     
1904     var object = objects[numberOfObjects - i];
1905     
1906 
1907     if (object) {
1908       // we have a valid object
1909       
1910       // check visibility
1911       if (!object['_visible']) {
1912         
1913         // not visible, continue to the next one..
1914         continue;
1915         
1916       }
1917       
1918       var id = object['_id'];
1919       
1920       var magicMode = object['_magicMode'];
1921       
1922       var vertexBuffer = this._vertexBuffers.get(id);
1923       var normalBuffer = this._normalBuffers.get(id);
1924       
1925       var colorBuffer = this._colorBuffers.get(id);
1926       var texturePositionBuffer = this._texturePositionBuffers.get(id);
1927       
1928       // ..bind the glBuffers
1929       
1930       // VERTICES
1931       this._gl.bindBuffer(this._gl.ARRAY_BUFFER, vertexBuffer._glBuffer);
1932       
1933       this._gl.vertexAttribPointer(aPosition, vertexBuffer._itemSize,
1934           this._gl.FLOAT, false, 0, 0);
1935       
1936       // NORMALS
1937       this._gl.bindBuffer(this._gl.ARRAY_BUFFER, normalBuffer._glBuffer);
1938       
1939       this._gl.vertexAttribPointer(aNormal, normalBuffer._itemSize,
1940           this._gl.FLOAT, false, 0, 0);
1941       
1942       if (picking) {
1943         
1944         // in picking mode, we use a color based on the id of this object
1945         this._gl.uniform1i(uUsePicking, true);
1946         
1947       } else {
1948         
1949         // in picking mode, we use a color based on the id of this object
1950         this._gl.uniform1i(uUsePicking, false);
1951         
1952       }
1953       
1954       // COLORS
1955       if (colorBuffer && !picking && !magicMode) {
1956         
1957         // point colors are defined for this object and there is not picking
1958         // request and no magicMode active
1959         
1960         // de-activate the useObjectColor flag on the shader
1961         this._gl.uniform1i(uUseObjectColor, false);
1962         
1963         this._gl.bindBuffer(this._gl.ARRAY_BUFFER, colorBuffer._glBuffer);
1964         
1965         this._gl.vertexAttribPointer(aColor, colorBuffer._itemSize,
1966             this._gl.FLOAT, false, 0, 0);
1967         
1968       } else {
1969         
1970         // we have a fixed object color or this is 'picking' mode
1971         var useObjectColor = 1;
1972         
1973         // some magic mode support
1974         if (magicMode && !picking) {
1975           
1976           useObjectColor = 0;
1977           
1978         }
1979         
1980         // activate the useObjectColor flag on the shader
1981         // in magicMode, this is always false!
1982         this._gl.uniform1i(uUseObjectColor, useObjectColor);
1983         
1984         var objectColor = object['_color'];
1985         
1986         if (picking) {
1987           
1988           if (id > 999) {
1989             
1990             throw new Error('Id out of bounds.');
1991             
1992           }
1993           
1994           // split the id
1995           // f.e. 15:
1996           // r = 0 / 10
1997           // g = 1 / 10
1998           // b = 5 / 10
1999           var r = Math.floor(id * 0.01);
2000           var g = Math.floor(id * 0.1) - r * 10;
2001           var b = id - r * 100 - g * 10;
2002           
2003           // and set it as the color
2004           objectColor = [r / 10, g / 10, b / 10];
2005         }
2006         
2007         this._gl.uniform3f(uObjectColor, parseFloat(objectColor[0]),
2008             parseFloat(objectColor[1]), parseFloat(objectColor[2]));
2009         
2010         // we always have to configure the attribute of the point colors
2011         // even if no point colors are in use
2012         this._gl.vertexAttribPointer(aColor, vertexBuffer._itemSize,
2013             this._gl.FLOAT, false, 0, 0);
2014         
2015       }
2016       
2017       // OPACITY
2018       this._gl.uniform1f(uObjectOpacity, parseFloat(object['_opacity']));
2019       
2020       // TEXTURE
2021       if (object._texture && texturePositionBuffer && !picking) {
2022         //
2023         // texture associated to this object
2024         //
2025         
2026         // activate the texture flag on the shader
2027         this._gl.uniform1i(uUseTexture, true);
2028         
2029         // setup the sampler
2030         
2031         // bind the texture
2032         this._gl.activeTexture(this._gl.TEXTURE0);
2033         
2034         // grab the texture from the internal hash map using the filename as the
2035         // key
2036         this._gl.bindTexture(this._gl.TEXTURE_2D, this._textures
2037             .get(object._texture['_id']));
2038         this._gl.uniform1i(uTextureSampler, 0);
2039         
2040         // propagate the current texture-coordinate-map to WebGL
2041         this._gl.bindBuffer(this._gl.ARRAY_BUFFER,
2042             texturePositionBuffer._glBuffer);
2043         
2044         this._gl.vertexAttribPointer(aTexturePosition,
2045             texturePositionBuffer._itemSize, this._gl.FLOAT, false, 0, 0);
2046         
2047       } else {
2048         
2049         // no texture for this object or 'picking' mode
2050         this._gl.uniform1i(uUseTexture, false);
2051         
2052         // we always have to configure the attribute of the texture positions
2053         // even if no textures are in use
2054         this._gl.vertexAttribPointer(aTexturePosition, vertexBuffer._itemSize,
2055             this._gl.FLOAT, false, 0, 0);
2056         
2057       }
2058       
2059       // VOLUMES
2060       // several special values need to be passed to the shaders if the object
2061       // is a X.slice (part of an X.volume)
2062       if (object instanceof X.slice && object._volume) {
2063         
2064         // we got a volume
2065         var volume = object._volume;
2066         
2067         // pass the lower threshold
2068         this._gl.uniform1f(uVolumeLowerThreshold, volume['_lowerThreshold']);
2069         // pass the upper threshold
2070         this._gl.uniform1f(uVolumeUpperThreshold, volume['_upperThreshold']);
2071         
2072         // pass the scalar range
2073         var scalarRange = volume._scalarRange;
2074         this._gl.uniform1f(uVolumeScalarMin, scalarRange[0]);
2075         this._gl.uniform1f(uVolumeScalarMax, scalarRange[1]);
2076         
2077         // opacity, only if volume rendering is active
2078         if (volume['_volumeRendering']) {
2079           
2080           this._gl.uniform1f(uObjectOpacity, parseFloat(volume['_opacity']));
2081           
2082         }
2083         
2084       }
2085       
2086       // TRANSFORMS
2087       // propagate transform to the uniform matrices of the shader
2088       this._gl.uniformMatrix4fv(uObjectTransform, false,
2089           object._transform._glMatrix);
2090       
2091       // POINT SIZE
2092       var pointSize = 1;
2093       if (object['_type'] == X.object.types.POINTS) {
2094         pointSize = object['_pointSize'];
2095       }
2096       this._gl.uniform1f(uPointSize, pointSize);
2097       
2098       //
2099       // .. and draw with the object's DRAW MODE
2100       //
2101       var drawMode = -1;
2102       if (object.type() == X.object.types.TRIANGLES) {
2103         
2104         drawMode = this._gl.TRIANGLES;
2105         if (statisticsEnabled) {
2106           trianglesCounter += (vertexBuffer._itemCount / 3);
2107         }
2108         
2109       } else if (object.type() == X.object.types.LINES) {
2110         
2111         this._gl.lineWidth(object.lineWidth());
2112         
2113         drawMode = this._gl.LINES;
2114         if (statisticsEnabled) {
2115           linesCounter += (vertexBuffer._itemCount / 2);
2116         }
2117         
2118       } else if (object.type() == X.object.types.POINTS) {
2119         
2120         drawMode = this._gl.POINTS;
2121         if (statisticsEnabled) {
2122           pointsCounter += vertexBuffer._itemCount;
2123         }
2124         
2125       } else if (object.type() == X.object.types.TRIANGLE_STRIPS) {
2126         
2127         drawMode = this._gl.TRIANGLE_STRIP;
2128         if (statisticsEnabled) {
2129           trianglesCounter += (vertexBuffer._itemCount / 3);
2130         }
2131         
2132       } else if (object.type() == X.object.types.POLYGONS) {
2133         
2134         // TODO right now, this is hacked.. we need to use the Van Gogh
2135         // triangulation algorithm or something faster to properly convert
2136         // POLYGONS to TRIANGLES.
2137         // Remark: The Van Gogh algorithm is implemented in the
2138         // X.object.toCSG/fromCSG functions but not used here.
2139         if (vertexBuffer._itemCount % 3 == 0) {
2140           
2141           drawMode = this._gl.TRIANGLES;
2142           
2143         } else {
2144           
2145           drawMode = this._gl.TRIANGLE_FAN;
2146           
2147         }
2148         
2149         if (statisticsEnabled) {
2150           trianglesCounter += (vertexBuffer._itemCount / 3);
2151         }
2152         
2153       }
2154       
2155       if (statisticsEnabled) {
2156         
2157         verticesCounter += vertexBuffer._itemCount;
2158         
2159       }
2160       
2161       // push it to the GPU, baby..
2162       this._gl.drawArrays(drawMode, 0, vertexBuffer._itemCount);
2163       
2164     }
2165     
2166   } while (--i); // loop through objects
2167   
2168   if (statisticsEnabled) {
2169     
2170     var statistics = "Objects: " + numberOfObjects + " | ";
2171     statistics += "Vertices: " + verticesCounter + " | ";
2172     statistics += "Triangles: " + Math.round(trianglesCounter) + " | ";
2173     statistics += "Lines: " + linesCounter + " | ";
2174     statistics += "Points: " + pointsCounter + " | ";
2175     statistics += "Textures: " + this._textures.getCount();
2176     window.console.log(statistics);
2177     
2178   }
