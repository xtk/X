 46 /**
 47  * Create a parser for .NRRD files.
 48  * 
 49  * @constructor
 50  * @extends X.parser
 51  */
 52 X.parserNRRD = function() {
 53 
 54   //
 55   // call the standard constructor of X.parser
 56   goog.base(this);
 57   
 58   //
 59   // class attributes
 60   
 61   /**
 62    * @inheritDoc
 63    * @const
 64    */
 65   this['_className'] = 'parserNRRD';
 66   
 67 };
 68 // inherit from X.parser
 69 goog.inherits(X.parserNRRD, X.parser);

 72 /**
 73  * @inheritDoc
 74  */
 75 X.parserNRRD.prototype.parse = function(object, data) {
 76 
 77   // the position in the file
 78   var position = 0;
 79   
 80   // grab the header
 81   var headerRegexMatch = data.match(/^([\s\S]*?)\r?\n\r?\n/);
 82   position = headerRegexMatch[0].length; // the one _with_ the blank line
 83   var header = headerRegexMatch[1]; // the one without the blank line
 84   
 85   // parse the header
 86   this.parseHeader(header);
 87   
 88   // now we have all kinds of things attached to this reader..
 89   // this was done by M. Lauer
 90   // I don't really like it but it works..
 91   
 92   var _data = 0; // the data without header
 93   
 94   if (this.encoding == 'gzip' || this.encoding == 'gz') {
 95     // we need to decompress the datastream
 96     _data = new JXG.Util.Unzip(data.substr(position)).unzip()[0][0];
 97   } else {
 98     // we can use the data directly
 99     _data = data.substr(position);
100   }
101   
102   var numberOfPixels = this.sizes[0] * this.sizes[1] * this.sizes[2];
103   
104   //
105   // parse the (unzipped) data to a datastream of the correct type
106   //
107   var datastream = new Array(numberOfPixels);
108   
109   // we store the min and max values to be able to convert the values to uint8
110   // later
111   var max = -Infinity;
112   var min = Infinity;
113   
114   var i;
115   for (i = 0; i < numberOfPixels; i++) {
116     // parseFunc was defined by analyzing the nrrd header
117     var pixelValue = this.parseFunc(_data, 0 + (i * this.parseBytes));
118     datastream[i] = pixelValue;
119     max = Math.max(max, pixelValue);
120     min = Math.min(min, pixelValue);
121   }
122   
123   //
124   // we know enough to create the object
125   
126   // origin
127   // var _origin = this['space origin'];
128   // _origin = _origin.slice(1, _origin.length).split(',');
129   // object._center[0] = parseFloat(_origin[0]) - this.sizes[0] / 2;
130   // object._center[1] = parseFloat(_origin[1]) - this.sizes[1] / 2;
131   // object._center[2] = parseFloat(_origin[2]) - this.sizes[2] / 2;
132   
133   // dimensions
134   object._dimensions = [this.sizes[0], this.sizes[1], this.sizes[2]];
135   
136   // spacing
137   var spacingX = new goog.math.Vec3(this.vectors[0][0], this.vectors[0][1],
138       this.vectors[0][2]).magnitude();
139   var spacingY = new goog.math.Vec3(this.vectors[1][0], this.vectors[1][1],
140       this.vectors[1][2]).magnitude();
141   var spacingZ = new goog.math.Vec3(this.vectors[2][0], this.vectors[2][1],
142       this.vectors[2][2]).magnitude();
143   object._spacing = [spacingX, spacingY, spacingZ];
144   
145 
146   // attach the scalar range to the volume
147   object._scalarRange = [min, max];
148   // .. and set the default threshold
149   object.threshold(min, max);
150   
151   // create the object
152   object.create_();
153   
154   // now we have the values and need to reslice in the 3 orthogonal directions
155   // and create the textures for each slice
156   this.reslice(object, datastream, this.sizes, min, max);
157   
158   // all done..
159   var modifiedEvent = new X.event.ModifiedEvent();
160   modifiedEvent._object = object;
161   this.dispatchEvent(modifiedEvent);
162   
163 };

166 X.parserNRRD.prototype.parseHeader = function(header) {
167 
168   var data, field, fn, i, l, lines, m, _i, _len, _results;
169   lines = header.split(/\r?\n/);
170   for (_i = 0, _len = lines.length; _i < _len; _i++) {
171     l = lines[_i];
172     if (l.match(/NRRD\d+/)) {
173       this.isNrrd = true;
174     } else if (l.match(/^#/)) {
175     } else if (m = l.match(/(.*):(.*)/)) {
176       field = m[1].trim();
177       data = m[2].trim();
178       fn = this.fieldFunctions[field];
179       if (fn) {
180         fn.call(this, data);
181       } else {
182         this[field] = data;
183       }
184     }
185   }
186   if (!this.isNrrd) {
187     throw new Error('Not an NRRD file');
188   }
189   if (this.encoding !== 'raw' && this.encoding !== 'gzip' &&
190       this.encoding !== 'gz') {
191     throw new Error('Only raw or gz/gzip encoding is allowed');
192   }
193   if (!this.vectors) {
194     this.vectors = [new goog.math.Vec3(1, 0, 0), new goog.math.Vec3(0, 1, 0),
195                     new goog.math.Vec3(0, 0, 1)];
196     if (this.spacings) {
197       _results = [];
198       for (i = 0; i <= 2; i++) {
199         _results.push(!isNaN(this.spacings[i]) ? this.vectors[i]
200             .scale(this.spacings[i]) : void 0);
201       }
202       return _results;
203     }
204   }
205 };

207 X.parserNRRD.prototype.fieldFunctions = {
208   'type': function(data) {
209 
210     switch (data) {
211     case 'unsigned char':
212     case 'uint8':
213       this.parseFunc = this.parseUChar8;
214       this.parseBytes = 1;
215       break;
216     case 'signed char':
217     case 'int8':
218       this.parseFunc = this.parseSChar8;
219       this.parseBytes = 1;
220       break;
221     case 'short':
222     case 'signed short':
223     case 'unsigned short':
224     case 'short int':
225     case 'int16':
226       this.parseFunc = this.parseUInt16;
227       this.parseBytes = 2;
228       break;
229     case 'int':
230     case 'int32':
231       break;
232     case 'float':
233       this.parseFunc = this.parseFloat32;
234       this.parseBytes = 4;
235       break;
236     default:
237       throw new Error('Only short/int/int8/float data is allowed. Found ' +
238           data);
239     }
240     return this.type = data;
241   },
242   'endian': function(data) {
243 
244     return this.endian = data;
245   },
246   'encoding': function(data) {
247 
248     return this.encoding = data;
249   },
250   'dimension': function(data) {
251 
252     return this.dim = parseInt(data, 10);
253   },
254   'sizes': function(data) {
255 
256     var i;
257     return this.sizes = (function() {
258 
259       var _i, _len, _ref, _results;
260       _ref = data.split(/\s+/);
261       _results = [];
262       for (_i = 0, _len = _ref.length; _i < _len; _i++) {
263         i = _ref[_i];
264         _results.push(parseInt(i, 10));
265       }
266       return _results;
267     })();
268   },
269   'space directions': function(data) {
270 
271     var f, parts, v;
272     parts = data.match(/\(.*?\)/g);
273     return this.vectors = (function() {
274 
275       var _i, _len, _results;
276       _results = [];
277       for (_i = 0, _len = parts.length; _i < _len; _i++) {
278         v = parts[_i];
279         _results.push((function() {
280 
281           var _j, _len2, _ref, _results2;
282           _ref = v.slice(1, -1).split(/,/);
283           _results2 = [];
284           for (_j = 0, _len2 = _ref.length; _j < _len2; _j++) {
285             f = _ref[_j];
286             _results2.push(parseFloat(f));
287           }
288           return _results2;
289         })());
290       }
291       return _results;
292     })();
293   },
294   'spacings': function(data) {
295 
296     var f, parts;
297     parts = data.split(/\s+/);
298     return this.spacings = (function() {
299 
300       var _i, _len, _results;
301       _results = [];
302       for (_i = 0, _len = parts.length; _i < _len; _i++) {
303         f = parts[_i];
304         _results.push(parseFloat(f));
305       }
306       return _results;
307     })();
308   }
309 };
