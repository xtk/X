 30 // provides
 31 goog.provide('X.progressbar');

 33 // requires
 34 goog.require('X.base');
 35 goog.require('X.exception');
 36 goog.require('goog.dom');
 37 goog.require('goog.ui.ProgressBar');

 40 /**
 41  * Create a progress bar.
 42  * 
 43  * @constructor
 44  * @param {!Element} parent The parent element in the DOM tree.
 45  * @param {!number} initialvalue An optional initial value for this progress
 46  *          bar.
 47  * @extends goog.ui.ProgressBar
 48  */
 49 X.progressbar = function(parent, initialvalue) {
 50 
 51   // check if we have a valid parent
 52   if (!goog.isDefAndNotNull(parent)) {
 53     
 54     throw new X.exception('No valid parent element!');
 55     
 56   }
 57   
 58   // check if we have an initial value
 59   if (!goog.isDefAndNotNull(initialvalue)) {
 60     
 61     throw new X.exception('Invalid initial value!');
 62     
 63   }
 64   
 65   //
 66   // call the standard constructor of goog.ui.ProgressBar
 67   goog.base(this);
 68   
 69   //
 70   // class attributes
 71   
 72   /**
 73    * @inheritDoc
 74    * @const
 75    */
 76   this._className = 'progressbar';
 77   
 78   /**
 79    * The parent element in the DOM tree of this progress bar.
 80    * 
 81    * @type {!Element}
 82    * @protected
 83    */
 84   this._parent = parent;
 85   
 86   /**
 87    * The element for the CSS style of this progress bar.
 88    * 
 89    * @type {?Element}
 90    * @protected
 91    */
 92   this._style = null;
 93   
 94   /**
 95    * A green, full progressBar to replace this one once it's done.
 96    * 
 97    * @type {?X.progressbar}
 98    * @protected
 99    */
100   this._doneProgressBar = null;
101   
102   /**
103    * The collection of CSS definitions.
104    * 
105    * @type {!Array}
106    * @protected
107    */
108   this._css = [];
109   // configure some styles
110   var css1 = '.progress-bar-horizontal {\n';
111   css1 += '  position: relative;\n';
112   css1 += '  border: 1px solid #949dad;\n';
113   css1 += '  background: white;\n';
114   css1 += '  padding: 1px;\n';
115   css1 += '  overflow: hidden;\n';
116   css1 += '  margin: 2px;\n';
117   css1 += '  width: 100px;\n';
118   css1 += '  height: 5px;\n';
119   css1 += '}';
120   var css2 = '.progress-bar-thumb {\n';
121   css2 += '  position: relative;\n';
122   css2 += '  background: #F62217;\n';
123   css2 += '  overflow: hidden;\n';
124   css2 += '  width: 0%;\n';
125   css2 += '  height: 100%;\n';
126   css2 += '}';
127   var css3 = '.progress-bar-thumb-done {\n';
128   css3 += '  background: #57E964;\n';
129   css3 += '}';
130   this._css = [css1, css2, css3];
131   
132   // set initial value
133   this.setValue(initialvalue);
134   
135   // now initialize
136   this.init_();
137   
138 };
139 // inherit from X.base
140 goog.inherits(X.progressbar, goog.ui.ProgressBar);
141 
142 /**
143  * Internal function to initialize the progress bar which gets called by the
144  * constructor.
145  * 
146  * @private
147  */
148 X.progressbar.prototype.init_ = function() {
149 
150   // enable relative positioning for the main container
151   // this is required to place the progressBar in the center
152   this._parent.style.position = 'relative';
153   
154   //
155   // apply CSS styles to the document
156   //
157   var head = goog.dom.getDocument().getElementsByTagName('head')[0];
158   var style = goog.dom.createDom('style');
159   style.type = 'text/css';
160   style.media = 'screen';
161   var css = goog.dom.createTextNode(this._css[0]);
162   var css2 = goog.dom.createTextNode(this._css[1]);
163   var css3 = goog.dom.createTextNode(this._css[2]);
164   goog.dom.appendChild(head, style);
165   goog.dom.appendChild(style, css);
166   goog.dom.appendChild(style, css2);
167   goog.dom.appendChild(style, css3);
168   // save the style element for later
169   this._style = style;
170   
171   // render..
172   this.render(this._parent);
173   
174   // place the progressBar in the center
175   var pbElement = this.getElement();
176   pbElement.style.position = 'absolute';
177   pbElement.style.top = (this._parent.clientHeight - 5) / 2;
178   pbElement.style.left = (this._parent.clientWidth - 100) / 2;
179   
180 };

183 /**
184  * Display a green, full progress bar.
185  */
186 X.progressbar.prototype.done = function() {
187 
188   // save the position of the original progress bar
189   var top = this.getElement().style.top;
190   var left = this.getElement().style.left;
191   
192   // .. destroy the original progress bar (on the DOM side)
193   goog.dom.removeNode(this.getElement());
194   
195   // create a 'fake' new progress bar in place of the original one
196   // the new one always shows 100
197   var pb = new X.progressbar(this._parent, 100);
198   
199   // place the progressBar in the center
200   var pbElement = pb.getElement();
201   pbElement.style.position = 'absolute';
202   pbElement.style.top = top;
203   pbElement.style.left = left;
204   
205   // get the actual progress element of the progressBar
206   var pbBar = goog.dom.getFirstElementChild(pbElement);
207   
208   // .. change the color to green
209   pbBar.classList.add('progress-bar-thumb-done');
210   
211   this._doneProgressBar = pb;
212   
213 };
214 
215 
216 /**
217  * Remove the progress bar from the document.
218  */
219 X.progressbar.prototype.kill = function() {
220 
221   // hide the progress bar
222   if (this._style) {
223     goog.dom.removeNode(this._style);
224   }
225   if (this.getElement()) {
226     goog.dom.removeNode(this.getElement());
227   }
228   if (this._doneProgressBar) {
229     goog.dom.removeNode(this._doneProgressBar.getElement());
230   }
231   
232   this._style = null;
233   this._doneProgressBar = null;
234   
235 };
