 79 /**
 80  * Process a numerical array and calculate some basic stats: o mean o variance o
 81  * deviation o prod o sum o min; minIndex o max; maxIndex
 82  * 
 83  * @param {*} data The numerical data array to process.
 84  */
 85 X.parser.prototype.stats_calc = function(data) {
 86 
 87   var r = {
 88     mean: 0,
 89     variance: 0,
 90     deviation: 0,
 91     prod: 1,
 92     sum: 0,
 93     min: 0,
 94     minIndex: 0,
 95     max: 0,
 96     maxIndex: 0
 97   };
 98   var t = data.length;
 99   r.size = t;
100   for ( var m = 0, p = 1, s = 0, l = t; l >= 0; l--) {
101     s += data[l];
102     p *= data[l];
103     if (r.min >= data[l]) {
104       r.min = data[l];
105       r.minIndex = l;
106     }
107     if (r.max <= data[l]) {
108       r.max = data[l];
109       r.maxIndex = l;
110     }
111   }
112   ;
113   r.prod = p;
114   r.sum = s;
115   for (m = r.mean = s / t, l = t, s = 0; l--; s += Math.pow(data[l] - m, 2)) {
116     ;
117   }
118   return r.deviation = Math.sqrt(r.variance = s / t), r;
119 };

131 X.parser.prototype.parseFloat32 = function(data, offset) {
132 
133   var b3 = this.parseUChar8(data, offset), b2 = this.parseUChar8(data,
134       offset + 1), b1 = this.parseUChar8(data, offset + 2), b0 = this
135       .parseUChar8(data, offset + 3),
136 
137   sign = 1 - (2 * (b0 >> 7)), exponent = (((b0 << 1) & 0xff) | (b1 >> 7)) - 127, mantissa = ((b1 & 0x7f) << 16) |
138       (b2 << 8) | b3;
139   
140   if (mantissa == 0 && exponent == -127) {
141     return 0.0;
142   }
143   
144   return sign * (1 + mantissa * Math.pow(2, -23)) * Math.pow(2, exponent);
145   
146 };

149 X.parser.prototype.parseFloat32EndianSwapped = function(data, offset) {
150 
151   var b0 = this.parseUChar8(data, offset), b1 = this.parseUChar8(data,
152       offset + 1), b2 = this.parseUChar8(data, offset + 2), b3 = this
153       .parseUChar8(data, offset + 3),
154 
155   sign = 1 - (2 * (b0 >> 7)), exponent = (((b0 << 1) & 0xff) | (b1 >> 7)) - 127, mantissa = ((b1 & 0x7f) << 16) |
156       (b2 << 8) | b3;
157   
158   if (mantissa == 0 && exponent == -127) {
159     return 0.0;
160   }
161   
162   return sign * (1 + mantissa * Math.pow(2, -23)) * Math.pow(2, exponent);
163   
164 };

167 X.parser.prototype.parseFloat32Array = function(data, offset, elements) {
168 
169   var arr = new Array();
170   
171   var max = 0;
172   var min = Infinity;
173   
174   var i;
175   for (i = 0; i < elements; i++) {
176     var val = this.parseFloat32(data, offset + (i * 4));
177     arr[i] = val;
178     max = Math.max(max, val);
179     min = Math.min(min, val);
180   }
181   
182   return [arr, max, min];
183 };

206 X.parser.prototype.parseUInt32 = function(data, offset) {
207 
208   var b0 = this.parseUChar8(data, offset), b1 = this.parseUChar8(data,
209       offset + 1), b2 = this.parseUChar8(data, offset + 2), b3 = this
210       .parseUChar8(data, offset + 3);
211   
212   return (b3 << 24) + (b2 << 16) + (b1 << 8) + b0;

314 
315   var b = this.parseUChar8(data, offset);
316   return b > 127 ? b - 256 : b;

345 X.parser.prototype.reslice = function(object, datastream, sizes, min, max) {
346 
347   // number of slices in scan direction
348   var slices = sizes[2];
349   // number of rows in each slice in scan direction
350   var rowsCount = sizes[1];
351   // number of cols in each slice in scan direction
352   var colsCount = sizes[0];
353   
354   // do we have a labelMap?
355   var hasLabelMap = object._labelMap != null;
356   
357   // slice dimensions in scan direction
358   var numberPixelsPerSlice = rowsCount * colsCount;
359   
360   // allocate 3d image array [slices][rows][cols]
361   var image = new Array(slices);
362   for ( var iS = 0; iS < slices; iS++) {
363     image[iS] = new Array(rowsCount);
364     for ( var iR = 0; iR < rowsCount; iR++) {
365       image[iS][iR] = new Array(colsCount);
366     }
367   }
368   
369   var pixelValue = 0;
370   
371   // loop through all slices in scan direction
372   //
373   // this step creates the slices in X-direction and fills the 3d image array at
374   // the same time
375   // combining the two operations saves some time..
376   var z = 0;
377   for (z = 0; z < slices; z++) {
378     
379     // grab the pixels for the current slice z
380     var currentSlice = datastream.slice(z * (numberPixelsPerSlice), (z + 1) *
381         numberPixelsPerSlice);
382     // the texture has 3 times the pixel value + 1 opacity value for all pixels
383     var textureForCurrentSlice = new Uint8Array(4 * numberPixelsPerSlice);
384     
385     // now loop through all pixels of the current slice
386     var row = 0;
387     var col = 0;
388     var p = 0; // just a counter
389     
390     for (row = 0; row < rowsCount; row++) {
391       for (col = 0; col < colsCount; col++) {
392         
393         // map pixel values
394         pixelValue = currentSlice[p];
395         var pixelValue_r = 0;
396         var pixelValue_g = 0;
397         var pixelValue_b = 0;
398         var pixelValue_a = 0;
399         if (object._colorTable) {
400           // color table!
401           var lookupValue = object._colorTable._map.get(Math.floor(pixelValue));
402           
403           // check for out of range and use the last label value in this case
404           if (!lookupValue) {
405             lookupValue = object._colorTable._map.get(object._colorTable._map
406                 .getCount() - 1);
407           }
408           
409           pixelValue_r = 255 * lookupValue[1];
410           pixelValue_g = 255 * lookupValue[2];
411           pixelValue_b = 255 * lookupValue[3];
412           pixelValue_a = 255 * lookupValue[4];
413         } else {
414           // no color table, 1-channel gray value
415           pixelValue_r = pixelValue_g = pixelValue_b = 255 * (pixelValue / max);
416           pixelValue_a = 255;
417         }
418         
419         var textureStartIndex = p * 4;
420         textureForCurrentSlice[textureStartIndex] = pixelValue_r;
421         textureForCurrentSlice[++textureStartIndex] = pixelValue_g;
422         textureForCurrentSlice[++textureStartIndex] = pixelValue_b;
423         textureForCurrentSlice[++textureStartIndex] = pixelValue_a;
424         
425         // save the pixelValue in the 3d image data
426         image[z][row][col] = pixelValue;
427         
428         p++;
429         
430       }
431       
432     }
433     
434     // create the texture for slices in X-direction
435     var pixelTexture = new X.texture();
436     pixelTexture.setRawData(textureForCurrentSlice);
437     pixelTexture.setRawDataWidth(colsCount);
438     pixelTexture.setRawDataHeight(rowsCount);
439     
440     currentSlice = object._slicesZ.children()[z];
441     currentSlice.setTexture(pixelTexture);
442     if (hasLabelMap) {
443       
444       // if this object has a labelMap,
445       // we have it loaded at this point (for sure)
446       // ..so we can attach it as the second texture to this slice
447       currentSlice._labelMap = object._labelMap._slicesZ.children()[z]
448           .texture();
449       
450     }
451     
452   }
453   
454   // the following parses the 3d image array according to the Y- and the
455   // Z-direction of the slices
456   // this was unrolled for more performance
457   
458   // for Y-direction
459   // all slices are along the rows of the image
460   // all rows are along the slices of the image
461   // all cols are along the cols of the image
462   //  
463   for (row = 0; row < rowsCount; row++) {
464     
465     var textureForCurrentSlice = new Uint8Array(4 * slices * colsCount);
466     var p = 0; // just a counter
467     for (z = 0; z < slices; z++) {
468       for (col = 0; col < colsCount; col++) {
469         
470         pixelValue = image[z][row][col];
471         var pixelValue_r = 0;
472         var pixelValue_g = 0;
473         var pixelValue_b = 0;
474         var pixelValue_a = 0;
475         if (object._colorTable) {
476           // color table!
477           var lookupValue = object._colorTable._map.get(Math.floor(pixelValue));
478           
479           // check for out of range and use the last label value in this case
480           if (!lookupValue) {
481             lookupValue = object._colorTable._map.get(object._colorTable._map
482                 .getCount() - 1);
483           }
484           
485           pixelValue_r = 255 * lookupValue[1];
486           pixelValue_g = 255 * lookupValue[2];
487           pixelValue_b = 255 * lookupValue[3];
488           pixelValue_a = 255 * lookupValue[4];
489         } else {
490           // no color table, 1-channel gray value
491           pixelValue_r = pixelValue_g = pixelValue_b = 255 * (pixelValue / max);
492           pixelValue_a = 255;
493         }
494         
495         var textureStartIndex = p * 4;
496         textureForCurrentSlice[textureStartIndex] = pixelValue_r;
497         textureForCurrentSlice[++textureStartIndex] = pixelValue_g;
498         textureForCurrentSlice[++textureStartIndex] = pixelValue_b;
499         textureForCurrentSlice[++textureStartIndex] = pixelValue_a;
500         
501         p++;
502         
503       }
504     }
505     
506     var pixelTexture = new X.texture();
507     pixelTexture.setRawData(textureForCurrentSlice);
508     pixelTexture.setRawDataWidth(colsCount);
509     pixelTexture.setRawDataHeight(slices);
510     
511     currentSlice = object._slicesY.children()[row];
512     currentSlice.setTexture(pixelTexture);
513     if (hasLabelMap) {
514       
515       // if this object has a labelMap,
516       // we have it loaded at this point (for sure)
517       // ..so we can attach it as the second texture to this slice
518       currentSlice._labelMap = object._labelMap._slicesY.children()[row]
519           .texture();
520       
521     }
522     
523   }
524   
525 
526   // for Z
527   // all slices are along the cols of the image
528   // all rows are along the slices of the image
529   // all cols are along the rows of the image
530   //  
531   for (col = 0; col < colsCount; col++) {
532     var textureForCurrentSlice = new Uint8Array(4 * slices * rowsCount);
533     var p = 0; // just a counter
534     for (z = 0; z < slices; z++) {
535       for (row = 0; row < rowsCount; row++) {
536         
537         pixelValue = image[z][row][col];
538         var pixelValue_r = 0;
539         var pixelValue_g = 0;
540         var pixelValue_b = 0;
541         var pixelValue_a = 0;
542         if (object._colorTable) {
543           // color table!
544           var lookupValue = object._colorTable._map.get(Math.floor(pixelValue));
545           
546           // check for out of range and use the last label value in this case
547           if (!lookupValue) {
548             lookupValue = object._colorTable._map.get(object._colorTable._map
549                 .getCount() - 1);
550           }
551           
552           pixelValue_r = 255 * lookupValue[1];
553           pixelValue_g = 255 * lookupValue[2];
554           pixelValue_b = 255 * lookupValue[3];
555           pixelValue_a = 255 * lookupValue[4];
556         } else {
557           // no color table, 1-channel gray value
558           pixelValue_r = pixelValue_g = pixelValue_b = 255 * (pixelValue / max);
559           pixelValue_a = 255;
560         }
561         
562         var textureStartIndex = p * 4;
563         textureForCurrentSlice[textureStartIndex] = pixelValue_r;
564         textureForCurrentSlice[++textureStartIndex] = pixelValue_g;
565         textureForCurrentSlice[++textureStartIndex] = pixelValue_b;
566         textureForCurrentSlice[++textureStartIndex] = pixelValue_a;
567         
568         p++;
569         
570       }
571     }
572     
573     var pixelTexture = new X.texture();
574     pixelTexture.setRawData(textureForCurrentSlice);
575     pixelTexture.setRawDataWidth(rowsCount);
576     pixelTexture.setRawDataHeight(slices);
577     
578     currentSlice = object._slicesX.children()[col];
579     currentSlice.setTexture(pixelTexture);
580     if (hasLabelMap) {
581       
582       // if this object has a labelMap,
583       // we have it loaded at this point (for sure)
584       // ..so we can attach it as the second texture to this slice
585       currentSlice._labelMap = object._labelMap._slicesX.children()[col]
586           .texture();
587       
588     }
589     
590   }
591   
592 };
