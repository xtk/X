 40 /**
 41  * Create a transform.
 42  * 
 43  * @constructor
 44  * @extends X.base
 45  */
 46 X.transform = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this['_className'] = 'transform';
 60   
 61   /**
 62    * The transformation matrix.
 63    * 
 64    * @type {!X.matrix}
 65    * @protected
 66    */
 67   this._matrix = X.matrix.createIdentityMatrix(4);
 68   
 69   /**
 70    * The transformation matrix as a 'ready-to-use'-gl version.
 71    * 
 72    * @type {!Object}
 73    * @protected
 74    */
 75   this._glMatrix = new Float32Array(this._matrix.flatten());
 76   
 77 };
 78 // inherit from X.base
 79 goog.inherits(X.transform, X.base);

 82 /**
 83  * Rotate around the X-axis.
 84  * 
 85  * @param {number} angle The angle to rotate in degrees.
 86  * @throws {Error} An exception, if the given angle is invalid.
 87  */
 88 X.transform.prototype.rotateX = function(angle) {
 89 
 90   if (!goog.isNumber(angle) || angle < -360 || angle > 360) {
 91     
 92     throw new Error('Invalid angle.');
 93     
 94   }
 95   
 96   var angleInRadii = angle * Math.PI / 180;
 97   
 98   this._matrix = this._matrix.rotate(angleInRadii, new goog.math.Vec3(0, 1, 0));
 99   this._glMatrix = new Float32Array(this._matrix.flatten());
100   
101   this._dirty = true;
102   
103 };

238 /**
239  * Set the transformation matrix.
240  * 
241  * @param {!X.matrix} matrix The transformation matrix.
242  */
243 X.transform.prototype.setMatrix = function(matrix) {
244 
245   var matrix_ = X.matrix.createIdentityMatrix(4);
246   
247   if (goog.isDefAndNotNull(matrix) && (matrix instanceof X.matrix)) {
248     
249     matrix_ = matrix;
250     
251   }
252   
253   this._matrix = matrix_;
254   this._glMatrix = new Float32Array(this._matrix.flatten());
255   
256   this._dirty = true;
257   
258 };

273 /**
274  * Flip the matrix value at the given index.
275  * 
276  * @param {number} i The row index.
277  * @param {number} j The column index.
278  * @private
279  */
280 X.transform.prototype.flip_ = function(i, j) {
281 
282   var oldValue = this._matrix.getValueAt(i, j);
283   if (!oldValue) {
284     oldValue = 0;
285   }
286   
287   this._matrix.setValueAt(i, j, oldValue * -1);
288   this._glMatrix = new Float32Array(this._matrix.flatten());
289   
290   this._dirty = true;
291   
292 };
