 36 // requires
 37 goog.require('X.event');
 38 goog.require('X.object');
 39 goog.require('X.parser');
 40 goog.require('X.triplets');
 41 goog.require('goog.math.Vec3');

 44 
 45 /**
 46  * Create a parser for the binary freesurfer meshes
 47  * 
 48  * @constructor
 49  * @extends X.parser
 50  */
 51 X.parserFSM = function() {
 52 
 53   //
 54   // call the standard constructor of X.parser
 55   goog.base(this);
 56   
 57   //
 58   // class attributes
 59   
 60   /**
 61    * @inheritDoc
 62    * @const
 63    */
 64   this['_className'] = 'parserFSM';
 65   
 66 };
 67 // inherit from X.parser
 68 goog.inherits(X.parserFSM, X.parser);
 69 
 70 
 71 /**
 72  * @inheritDoc
 73  */
 74 X.parserFSM.prototype.parse = function(object, data) {
 75 
 76   var p = object.points();
 77   var n = object.normals();
 78   
 79   // in .FSM files, the points are not ordered for rendering, so we need to
 80   // buffer everything in X.triplets containers and then order it
 81   var unorderedPoints = new X.triplets();
 82   
 83   var currentOffset = 0;
 84   
 85   // parse some magic bytes
 86   this.parseString(data, currentOffset, 3);
 87   currentOffset += 3;
 88   
 89   // Go through two newlines
 90   var iters = 0;
 91   var curChar;
 92   do {
 93     curChar = this.parseUChar8(data, currentOffset++);
 94     iters++;
 95   } while ((iters < 200) && (curChar != 0x0A))
 96 
 97   // Read one more newline
 98   curChar = this.parseUChar8(data, currentOffset++);
 99   
100   // get the number of vertices
101   var numberOfVertices = this.parseUInt32EndianSwapped(data, currentOffset);
102   currentOffset += 4;
103   
104   // get the number of triangles
105   var numberOfTriangles = this.parseUInt32EndianSwapped(data, currentOffset);
106   currentOffset += 4;
107   
108   // we count the appearance of indices to be able to average the normals
109   var indexCounter = [];
110   
111   // parse the coordinates from the data
112   var v = 0;
113   for (v = 0; v < numberOfVertices; v++) {
114     
115     // grab 3 floats which are X,Y,Z in world coordinates
116     var x = this.parseFloat32EndianSwapped(data, currentOffset);
117     currentOffset += 4;
118     var y = this.parseFloat32EndianSwapped(data, currentOffset);
119     currentOffset += 4;
120     var z = this.parseFloat32EndianSwapped(data, currentOffset);
121     currentOffset += 4;
122     
123     // initialize the index counter with 0
124     indexCounter[v] = 0;
125     
126     // store the points unordered
127     unorderedPoints.add(x, y, z);
128     
129   }
130   
131   // a list of triangles composed by indices
132   var triangles = [];
133   
134   // a list of normals stored by index
135   var normals = [];
136   
137   // parse the indices from the data
138   var t;
139   for (t = 0; t < numberOfTriangles; t++) {
140     
141     // grab the three indices which define a triangle
142     var index1 = this.parseUInt32EndianSwapped(data, currentOffset);
143     currentOffset += 4;
144     var index2 = this.parseUInt32EndianSwapped(data, currentOffset);
145     currentOffset += 4;
146     var index3 = this.parseUInt32EndianSwapped(data, currentOffset);
147     currentOffset += 4;
148     
149     // we count the appearance of the indices
150     indexCounter[index1] += 1;
151     indexCounter[index2] += 1;
152     indexCounter[index3] += 1;
153     
154     triangles.push([index1, index2, index3]);
155     normals.push(new goog.math.Vec3(0, 0, 0));
156   }
157   
158   // compute the triangle normals using the indices
159   for (t = 0; t < numberOfTriangles; t++) {
160     
161     var index1 = triangles[t][0];
162     var index2 = triangles[t][1];
163     var index3 = triangles[t][2];
164     
165     // create a triangle with the 3 vertices
166     var v1 = unorderedPoints.get(index1);
167     var v2 = unorderedPoints.get(index2);
168     var v3 = unorderedPoints.get(index3);
169     
170     //
171     // compute the normals
172     var v1v = new goog.math.Vec3(v1[0], v1[1], v1[2]);
173     var v2v = new goog.math.Vec3(v2[0], v2[1], v2[2]);
174     var v3v = new goog.math.Vec3(v3[0], v3[1], v3[2]);
175     
176     var n1 = v2v.clone().subtract(v1v);
177     var n2 = v3v.clone().subtract(v1v);
178     
179     var normal = goog.math.Vec3.cross(n1, n2).normalize();
180     
181     normals[index1] = normals[index1].add(normal);
182     normals[index2] = normals[index2].add(normal);
183     normals[index3] = normals[index3].add(normal);
184     
185   }
186   
187   // configure the points and normals for the X.object
188   for (t = 0; t < numberOfTriangles; t++) {
189     
190     var index1 = triangles[t][0];
191     var index2 = triangles[t][1];
192     var index3 = triangles[t][2];
193     
194     // create a triangle with the 3 vertices
195     var v1 = unorderedPoints.get(index1);
196     var v2 = unorderedPoints.get(index2);
197     var v3 = unorderedPoints.get(index3);
198     
199     // store the ordered vertices
200     p.add(v1[0], v1[1], v1[2]);
201     p.add(v2[0], v2[1], v2[2]);
202     p.add(v3[0], v3[1], v3[2]);
203     
204     // transform triangle normals to vertex normals
205     var normal1 = normals[index1].scale(1 / indexCounter[index1]).normalize();
206     var normal2 = normals[index2].scale(1 / indexCounter[index2]).normalize();
207     var normal3 = normals[index3].scale(1 / indexCounter[index3]).normalize();
208     
209     // .. add'em
210     n.add(normal1.x, normal1.y, normal1.z);
211     n.add(normal2.x, normal2.y, normal2.z);
212     n.add(normal3.x, normal3.y, normal3.z);
213     
214   }
215   
216   // .. and set the objectType to triangles
217   object.setType(X.object.types.TRIANGLES);
218   
219   // the object should be set up here, so let's fire a modified event
220   var modifiedEvent = new X.event.ModifiedEvent();
221   modifiedEvent._object = object;
222   this.dispatchEvent(modifiedEvent);
223   
224 };
