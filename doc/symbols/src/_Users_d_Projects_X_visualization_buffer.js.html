 30 // provides
 31 goog.provide('X.buffer');
 32 
 33 // requires
 34 goog.require('X.base');
 35 
 36 
 37 
 38 /**
 39  * Create a GL buffer container. Besides the actual GL buffer, the container
 40  * stores the number of items and the size of each item.
 41  * 
 42  * @constructor
 43  * @param {Object} glBuffer The GL buffer.
 44  * @param {number} itemCount The number of items.
 45  * @param {number} itemSize The size of each item.
 46  * @extends X.base
 47  */
 48 X.buffer = function(glBuffer, itemCount, itemSize) {
 49 
 50   if (!goog.isDefAndNotNull(glBuffer)) {
 51     
 52     throw new Error('Invalid GL Buffer.');
 53     
 54   }
 55   
 56   if (!goog.isDefAndNotNull(itemCount)) {
 57     
 58     throw new Error('Invalid number of items.');
 59     
 60   }
 61   
 62   if (!goog.isDefAndNotNull(itemSize)) {
 63     
 64     throw new Error('Invalid item size.');
 65     
 66   }
 67   
 68   // call the standard constructor of X.base
 69   goog.base(this);
 70   
 71   //
 72   // class attributes
 73   
 74   /**
 75    * @inheritDoc
 76    * @const
 77    */
 78   this['_className'] = 'buffer';
 79   
 80   /**
 81    * The GL buffer.
 82    * 
 83    * @type {Object}
 84    * @const
 85    */
 86   this._glBuffer = glBuffer;
 87   
 88   /**
 89    * The number of items.
 90    * 
 91    * @type {number}
 92    * @const
 93    */
 94   this._itemCount = itemCount;
 95   
 96   /**
 97    * The size of each item.
 98    * 
 99    * @type {number}
100    * @const
101    */
102   this._itemSize = itemSize;
103   
104 };
105 // inherit from X.base
106 goog.inherits(X.buffer, X.base);
