 41 /**
 42  * Create a parser for the ascii .VTK format.
 43  * 
 44  * @constructor
 45  * @extends X.parser
 46  */
 47 X.parserVTK = function() {
 48 
 49   //
 50   // call the standard constructor of X.base
 51   goog.base(this);
 52   
 53   //
 54   // class attributes
 55   
 56   /**
 57    * @inheritDoc
 58    * @const
 59    */
 60   this['_className'] = 'parserVTK';
 61   
 62 };
 63 // inherit from X.parser
 64 goog.inherits(X.parserVTK, X.parser);

 67 /**
 68  * @inheritDoc
 69  */
 70 X.parserVTK.prototype.parse = function(object, data) {
 71 
 72   var p = object.points();
 73   var n = object.normals();
 74   
 75   var dataAsArray = data.split('\n');
 76   var numberOfLines = dataAsArray.length;
 77   
 78   // in .VTK files, the points are not ordered for rendering, so we need to
 79   // buffer everything in X.triplets containers and then order it
 80   var unorderedPoints = new X.triplets();
 81   var unorderedNormals = new X.triplets();
 82   
 83   // .. we also need a buffer for all indices
 84   this._geometries = [];
 85   
 86   // even if vtk files support multiple object types in the same file, we only
 87   // support one kind
 88   this._objectType = X.object.types.TRIANGLES;
 89   
 90   // this mode indicates that the next lines will be X,Y,Z coordinates
 91   this._pointsMode = false;
 92   
 93   // this mode indicates that the next lines will be indices mapping to points
 94   // and pointData
 95   this._geometryMode = false;
 96   
 97   // this mode indicates that the next lines will be pointData
 98   this._pointDataMode = false;
 99   // one type of pointData are normals and right now the only supported ones
100   this._normalsMode = false;
101   
102 
103   //
104   // LOOP THROUGH ALL LINES
105   //
106   // This uses an optimized loop.
107   //
108   
109   /*
110    * Fast Duff's Device
111    * 
112    * @author Miller Medeiros <http://millermedeiros.com>
113    * 
114    * @version 0.3 (2010/08/25)
115    */
116   var i = 0;
117   var n2 = numberOfLines % 8;
118   while (n2--) {
119     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
120     i++;
121   }
122   
123   n2 = (numberOfLines * 0.125) ^ 0;
124   while (n2--) {
125     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
126     i++;
127     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
128     i++;
129     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
130     i++;
131     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
132     i++;
133     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
134     i++;
135     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
136     i++;
137     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
138     i++;
139     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
140     i++;
141   }
142   
143 
144   // now, configure the object according to the objectType
145   if (this._objectType == X.object.types.TRIANGLES) {
146     
147     this.configureTriangles(unorderedPoints, unorderedNormals, p, n);
148     
149   } else if (this._objectType == X.object.types.TRIANGLE_STRIPS) {
150     
151     this.configureTriangleStrips(unorderedPoints, unorderedNormals, p, n);
152     
153   } else if (this._objectType == X.object.types.LINES) {
154     
155     this.configureLines(unorderedPoints, unorderedNormals, p, n);
156     
157   } else if (this._objectType == X.object.types.POINTS) {
158     
159     this.configurePoints(unorderedPoints, unorderedNormals, p, n);
160     
161   } else if (this._objectType == X.object.types.POLYGONS) {
162     
163     this.configurePolygons(unorderedPoints, unorderedNormals, p, n);
164     
165   }
166   
167   // .. and set the objectType
168   object.setType(this._objectType);
169   
170   // the object should be set up here, so let's fire a modified event
171   var modifiedEvent = new X.event.ModifiedEvent();
172   modifiedEvent._object = object;
173   this.dispatchEvent(modifiedEvent);
174   
175 };

178 /**
179  * Parses a line of .VTK data and modifies the given X.triplets containers.
180  * 
181  * @param {!X.triplets} unorderedPoints A points container as a X.triplets
182  *          object.
183  * @param {!X.triplets} unorderedNormals A normals container as a X.triplets
184  *          object.
185  * @param {!string} line The line to parse.
186  * @protected
187  */
188 X.parserVTK.prototype.parseLine = function(unorderedPoints, unorderedNormals,
189     line) {
190 
191   // trim the line
192   line = line.replace(/^\s+|\s+$/g, '');
193   
194   // split to array
195   var lineFields = line.split(' ');
196   
197   // number of lineFields
198   var numberOfLineFields = lineFields.length;
199   
200   // the first field of the line can be a keyword to indicate different modes
201   var firstLineField = lineFields[0];
202   
203   // KEYWORD CHECK / MODE SWITCH
204   //
205   // identify the section of the next coming lines using the vtk keywords
206   if (firstLineField == 'POINTS') {
207     
208     // this means that real X,Y,Z points are coming
209     
210     this._pointsMode = true;
211     this._geometryMode = false;
212     this._pointDataMode = false;
213     
214     // go to next line
215     return;
216     
217   } else if (firstLineField == 'VERTICES') {
218     
219     // this means that triangles or points are coming
220     
221     this._geometryMode = true;
222     this._pointsMode = false;
223     this._pointDataMode = false;
224     
225     var numberOfElements = parseInt(lineFields[1], 10);
226     
227     if (numberOfElements >= 3) {
228       this._objectType = X.object.types.TRIANGLES;
229     } else if (numberOfElements == 1) {
230       this._objectType = X.object.types.POINTS;
231     } else {
232       
233       throw new Error('VTK file not supported!');
234       
235     }
236     
237     // reset all former geometries since we only support 1 geometry type per
238     // file (the last one specified)
239     this._geometries = [];
240     
241     // go to next line
242     return;
243     
244   } else if (firstLineField == 'TRIANGLE_STRIPS') {
245     
246     // this means that triangle_strips are coming
247     
248     this._geometryMode = true;
249     this._pointsMode = false;
250     this._pointDataMode = false;
251     this._objectType = X.object.types.TRIANGLE_STRIPS;
252     
253     // reset all former geometries since we only support 1 geometry type per
254     // file (the last one specified)
255     this._geometries = [];
256     
257     // go to next line
258     return;
259     
260   } else if (firstLineField == 'LINES') {
261     
262     // this means that lines are coming
263     
264     this._geometryMode = true;
265     this._pointsMode = false;
266     this._pointDataMode = false;
267     this._objectType = X.object.types.LINES;
268     
269     // reset all former geometries since we only support 1 geometry type per
270     // file (the last one specified)
271     this._geometries = [];
272     
273     // go to next line
274     return;
275     
276   } else if (firstLineField == 'POLYGONS') {
277     
278     // this means that polygons are coming
279     // we only support polygons which are triangles right now
280     
281     this._geometryMode = true;
282     this._pointsMode = false;
283     this._pointDataMode = false;
284     this._objectType = X.object.types.POLYGONS;
285     
286     // reset all former geometries since we only support 1 geometry type per
287     // file (the last one specified)
288     this._geometries = [];
289     
290     // go to next line
291     return;
292     
293   } else if (firstLineField == 'POINT_DATA') {
294     
295     // this means point-data is coming
296     // f.e. normals
297     
298     this._pointDataMode = true;
299     this._pointsMode = false;
300     this._geometryMode = false;
301     
302     // go to next line
303     return;
304     
305   }
306   
307   // PARSING
308   //
309   // now we parse according to the current mode
310   //
311   if (this._pointsMode) {
312     
313     // in pointsMode, each line has X,Y,Z coordinates separated by space
314     
315     if (numberOfLineFields == 1 || isNaN(parseFloat(firstLineField))) {
316       
317       // this likely means end of pointsMode
318       this._pointsMode = false;
319       
320       return;
321       
322     }
323     
324     // assume max. 9 coordinate values (== 3 points) in one row
325     if (numberOfLineFields >= 3) {
326       var x0 = parseFloat(lineFields[0]);
327       var y0 = parseFloat(lineFields[1]);
328       var z0 = parseFloat(lineFields[2]);
329       
330       unorderedPoints.add(x0, y0, z0);
331     }
332     
333     if (numberOfLineFields >= 6) {
334       var x1 = parseFloat(lineFields[3]);
335       var y1 = parseFloat(lineFields[4]);
336       var z1 = parseFloat(lineFields[5]);
337       unorderedPoints.add(x1, y1, z1);
338     }
339     
340     if (numberOfLineFields >= 9) {
341       var x2 = parseFloat(lineFields[6]);
342       var y2 = parseFloat(lineFields[7]);
343       var z2 = parseFloat(lineFields[8]);
344       
345       unorderedPoints.add(x2, y2, z2);
346     }
347     
348   } // end of pointsMode
349   else if (this._geometryMode) {
350     
351     // in geometryMode, each line has indices which map to points and pointsData
352     
353     if (numberOfLineFields == 1 || isNaN(parseFloat(firstLineField))) {
354       
355       // this likely means end of geometryMode
356       this._geometryMode = false;
357       return;
358       
359     }
360     
361     // the first element is the number of coming indices
362     // so we just slice the first element to get all indices
363     var values = lineFields.slice(1);
364     
365     // append all index values to the main geometries array
366     this._geometries.push(values);
367     
368   } // end of geometryMode
369   else if (this._pointDataMode) {
370     
371     // at the moment, only normals are supported as point-data
372     
373     if (firstLineField == 'NORMALS') {
374       
375       this._normalsMode = true;
376       
377       return;
378       
379     }
380     
381     if (numberOfLineFields == 1 || isNaN(parseFloat(firstLineField))) {
382       
383       // this likely means end of pointDataMode
384       this._pointDataMode = false;
385       this._normalsMode = false;
386       
387       return;
388       
389     }
390     
391     // the normals mode
392     if (this._normalsMode) {
393       
394       // assume 9 coordinate values (== 3 points) in one row
395       
396       if (numberOfLineFields >= 3) {
397         var x0 = parseFloat(lineFields[0]);
398         var y0 = parseFloat(lineFields[1]);
399         var z0 = parseFloat(lineFields[2]);
400         unorderedNormals.add(x0, y0, z0);
401       }
402       if (numberOfLineFields >= 6) {
403         var x1 = parseFloat(lineFields[3]);
404         var y1 = parseFloat(lineFields[4]);
405         var z1 = parseFloat(lineFields[5]);
406         unorderedNormals.add(x1, y1, z1);
407       }
408       if (numberOfLineFields >= 9) {
409         var x2 = parseFloat(lineFields[6]);
410         var y2 = parseFloat(lineFields[7]);
411         var z2 = parseFloat(lineFields[8]);
412         unorderedNormals.add(x2, y2, z2);
413       }
414       
415     } // end of normalsMode
416     
417   } // end of pointDataMode
418   
419 };

422 X.parserVTK.prototype.configureTriangles = function(unorderedPoints,
423     unorderedNormals, p, n) {
424 
425   // cache often used values for fast access
426   var numberOfUnorderedNormals = unorderedNormals.length();
427   
428   var numberOfGeometries = this._geometries.length;
429   var i = numberOfGeometries;
430   // we use this loop here since it's slightly faster than the for loop
431   do {
432     
433     // we want to loop through the geometries in the range 0..(N - 1)
434     var currentGeometry = this._geometries[numberOfGeometries - i];
435     var currentGeometryLength = currentGeometry.length;
436     
437     // in the sub-loop we loop through the indices of the current geometry
438     // since this is TRIANGLE_STRIPS rendering mode, we just add the point to
439     // the object's points
440     var k;
441     for (k = 0; k < currentGeometryLength; k++) {
442       //      
443       var currentIndex = parseInt(currentGeometry[k], 10);
444       
445       // grab the point with the currentIndex
446       var currentPoint = unorderedPoints.get(currentIndex);
447       
448       // .. and add it
449       p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
450       
451       if (currentIndex < numberOfUnorderedNormals) {
452         
453         // grab the normal with the currentIndex, if it exists
454         var currentNormals = unorderedNormals.get(currentIndex);
455         
456         // .. and add it
457         n.add(currentNormals[0], currentNormals[1], currentNormals[2]);
458         
459       } else {
460         
461         // add an artificial normal
462         // add an artificial normal
463         var artificialNormal = new goog.math.Vec3(currentPoint[0],
464             currentPoint[1], currentPoint[2]);
465         artificialNormal.normalize();
466         n.add(artificialNormal.x, artificialNormal.y, artificialNormal.z);
467         
468       }
469       
470     } // for loop through the currentGeometry
471     
472     i--;
473     
474   } while (i > 0);
475   
476 };

478 X.parserVTK.prototype.configureTriangleStrips = function(unorderedPoints,
479     unorderedNormals, p, n) {
480 
481   // cache often used values for fast access
482   var numberOfUnorderedNormals = unorderedNormals.length();
483   
484   var numberOfGeometries = this._geometries.length;
485   var i = numberOfGeometries;
486   // we use this loop here since it's slightly faster than the for loop
487   do {
488     
489     // we want to loop through the geometries in the range 0..(N - 1)
490     var currentGeometry = this._geometries[numberOfGeometries - i];
491     var currentGeometryLength = currentGeometry.length;
492     
493     // in the sub-loop we loop through the indices of the current geometry
494     // since this is TRIANGLE_STRIPS rendering mode, we add the first and the
495     // last element twice to interrupt the strips (as degenerated triangles)
496     // if we would not do this, then all strips would be connected
497     var k;
498     for (k = 0; k < currentGeometryLength; k++) {
499       //      
500       var currentIndex = parseInt(currentGeometry[k], 10);
501       
502       // grab the point with the currentIndex
503       var currentPoint = unorderedPoints.get(currentIndex);
504       
505       // .. and add it
506       p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
507       
508       if (k == 0 || k == currentGeometryLength - 1) {
509         
510         // if this is the first or last point of the triangle strip, add it
511         // again
512         
513         p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
514         
515       } // end of points
516       
517       if (currentIndex < numberOfUnorderedNormals) {
518         
519         // grab the normal with the currentIndex, if it exists
520         var currentNormals = unorderedNormals.get(currentIndex);
521         
522         // .. and add it
523         n.add(currentNormals[0], currentNormals[1], currentNormals[2]);
524         
525         if (k == 0 || k == currentGeometryLength - 1) {
526           
527           // if this is the first or last point of the triangle strip, add it
528           // again
529           n.add(currentNormals[0], currentNormals[1], currentNormals[2]);
530           
531         }
532         
533       } else {
534         
535         // add an artificial normal
536         // add an artificial normal
537         var artificialNormal = new goog.math.Vec3(currentPoint[0],
538             currentPoint[1], currentPoint[2]);
539         artificialNormal.normalize();
540         n.add(artificialNormal.x, artificialNormal.y, artificialNormal.z);
541         
542         if (k == 0 || k == currentGeometryLength - 1) {
543           
544           // if this is the first or last point of the triangle strip, add it
545           // again
546           n.add(artificialNormal.x, artificialNormal.y, artificialNormal.z);
547           
548         }
549         
550       } // end of normals
551       
552     } // for loop through the currentGeometry
553     
554     i--;
555     
556   } while (i > 0);
557   
558 };

622 /**
623  * @param unorderedPoints
624  * @param unorderedNormals
625  * @param p
626  * @param n
627  */
628 X.parserVTK.prototype.configureLines = function(unorderedPoints,
629     unorderedNormals, p, n) {
630 
631   // cache often used values for fast access
632   var numberOfUnorderedNormals = unorderedNormals.length();
633   
634   var numberOfGeometries = this._geometries.length;
635   var i = numberOfGeometries;
636   // we use this loop here since it's slightly faster than the for loop
637   do {
638     
639     // we want to loop through the geometries in the range 0..(N - 1)
640     var currentGeometry = this._geometries[numberOfGeometries - i];
641     var currentGeometryLength = currentGeometry.length;
642     
643     // in the sub-loop we loop through the indices of the current geometry
644     // since this is LINES rendering mode, we add the next element twice to
645     // interrupt the line segments (in webGL lines mode connects always 2
646     // points)
647     // if we would not do this, then all line segments would be connected
648     // therefore, the loop goes to (currentGeometryLength - 2)
649     var k;
650     for (k = 0; k < currentGeometryLength - 1; k++) {
651       //      
652       var currentIndex = parseInt(currentGeometry[k], 10);
653       var nextIndex = parseInt(currentGeometry[k + 1], 10);
654       
655       // grab the point with the currentIndex
656       var currentPoint = unorderedPoints.get(currentIndex);
657       // and connect it to the next one
658       var nextPoint = unorderedPoints.get(nextIndex);
659       
660       // .. and add both
661       p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
662       p.add(nextPoint[0], nextPoint[1], nextPoint[2]);
663       
664 
665       if (currentIndex < numberOfUnorderedNormals - 1) {
666         
667         // grab the normal with the currentIndex, if it exists
668         var currentNormals = unorderedNormals.get(currentIndex);
669         
670         // .. and also it's neighbor
671         var nextNormals = unorderedNormals.get(nextIndex);
672         
673         // .. and add both
674         n.add(currentNormals[0], currentNormals[1], currentNormals[2]);
675         n.add(nextNormals[0], nextNormals[1], nextNormals[2]);
676         
677       } else {
678         
679         // add an artificial normal
680         var artificialNormal = new goog.math.Vec3(currentPoint[0],
681             currentPoint[1], currentPoint[2]);
682         artificialNormal.normalize();
683         n.add(artificialNormal.x, artificialNormal.y, artificialNormal.z);
684         var artificialNormal2 = new goog.math.Vec3(nextPoint[0], nextPoint[1],
685             nextPoint[2]);
686         artificialNormal2.normalize();
687         n.add(artificialNormal2.x, artificialNormal2.y, artificialNormal2.z);
688         
689       }
690       
691     } // for loop through the currentGeometry
692     
693     i--;
694     
695   } while (i > 0);
696   
697 };
