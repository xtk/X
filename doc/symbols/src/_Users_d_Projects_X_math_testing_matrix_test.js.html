 29 function testXmatrixTranslate() {
 30 
 31   // create 4x4 identity matrix
 32   //
 33   // 1 0 0 0
 34   // 0 1 0 0
 35   // 0 0 1 0
 36   // 0 0 0 1
 37   //
 38   var matrix = X.matrix.createIdentityMatrix(4);
 39   
 40   // create 1x3 vector
 41   //
 42   // 6
 43   // 6
 44   // 6
 45   var vector = new goog.math.Vec3(6, 6, 6);
 46   
 47   // translating the matrix with the vector should result in this matrix
 48   //
 49   // 1 0 0 6
 50   // 0 1 0 6
 51   // 0 0 1 6
 52   // 0 0 0 1
 53   matrix = matrix.translate(vector);
 54   
 55   assertEquals(matrix.getValueAt(0, 0), 1);
 56   assertEquals(matrix.getValueAt(0, 1), 0);
 57   assertEquals(matrix.getValueAt(0, 2), 0);
 58   assertEquals(matrix.getValueAt(0, 3), 6);
 59   
 60   assertEquals(matrix.getValueAt(1, 0), 0);
 61   assertEquals(matrix.getValueAt(1, 1), 1);
 62   assertEquals(matrix.getValueAt(1, 2), 0);
 63   assertEquals(matrix.getValueAt(1, 3), 6);
 64   
 65   assertEquals(matrix.getValueAt(2, 0), 0);
 66   assertEquals(matrix.getValueAt(2, 1), 0);
 67   assertEquals(matrix.getValueAt(2, 2), 1);
 68   assertEquals(matrix.getValueAt(2, 3), 6);
 69   
 70   assertEquals(matrix.getValueAt(3, 0), 0);
 71   assertEquals(matrix.getValueAt(3, 1), 0);
 72   assertEquals(matrix.getValueAt(3, 2), 0);
 73   assertEquals(matrix.getValueAt(3, 3), 1);
 74   
 75   // TODO test also a 3x3 matrix translated by a 2d vector
 76

 79 function testXmatrixMultiplyByVector() {
 80 
 81   // create 4x4 identity matrix
 82   //
 83   // 1 0 0 0
 84   // 0 1 0 0
 85   // 0 0 1 0
 86   // 0 0 0 1
 87   //
 88   var matrix = X.matrix.createIdentityMatrix(4);
 89   
 90   // create a 1x3 vector
 91   var vector = new goog.math.Vec3(7, 8, 9);
 92   
 93   var baseline = new goog.math.Vec3(7, 8, 9);
 94   
 95   // multiplying the matrix by the vector should result in the vector since the
 96   // matrix is the identity
 97   vector = matrix.multiplyByVector(vector);
 98   
 99   assertEquals(vector.x, baseline.x);
100   assertEquals(vector.y, baseline.y);
101   assertEquals(vector.z, baseline.z);
102
