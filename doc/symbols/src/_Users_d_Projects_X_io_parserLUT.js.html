 40 /**
 41  * Create a parser for color maps (look-up tables).
 42  * 
 43  * @constructor
 44  * @extends X.parser
 45  */
 46 X.parserLUT = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this['_className'] = 'parserLUT';
 60   
 61 };
 62 // inherit from X.parser
 63 goog.inherits(X.parserLUT, X.parser);
 64 
 65 
 66 /**
 67  * @inheritDoc
 68  */
 69 X.parserLUT.prototype.parse = function(object, data, colorTable) {
 70 
 71   var dataAsArray = data.split('\n');
 72   
 73   var numberOfLines = dataAsArray.length;
 74   
 75   var i;
 76   for (i = 0; i < numberOfLines; ++i) {
 77     
 78     var line = dataAsArray[i];
 79     
 80     // trim the line
 81     line = line.replace(/^\s+|\s+$/g, '');
 82     
 83     // ignore comments
 84     if (line[0] == '#') {
 85       continue;
 86     }
 87     
 88     // split each line
 89     var lineFields = line.split(' ');
 90     
 91     // filter out multiple blanks
 92     lineFields = lineFields.filter(function(v) {
 93 
 94       return v != '';
 95       
 96     });
 97     
 98     // check if we have 6 values
 99     if (lineFields.length != 6) {
100       
101       // ignore this line
102       continue;
103       
104     }
105     
106     // here, we have a valid array containing
107     // labelValue, labelName, r, g, b, a
108     
109     // convert r, g, b, a to the range 0..1 and don't forget to make it a number
110     lineFields[2] = parseInt(lineFields[2], 10) / 255; // r
111     lineFields[3] = parseInt(lineFields[3], 10) / 255; // g
112     lineFields[4] = parseInt(lineFields[4], 10) / 255; // b
113     lineFields[5] = parseInt(lineFields[5], 10) / 255; // a
114     
115     // .. push it
116     colorTable.add(parseInt(lineFields[0], 10), lineFields[1], lineFields[2],
117         lineFields[3], lineFields[4], lineFields[5], 10);
118     
119   }
120   
121   var modifiedEvent = new X.event.ModifiedEvent();
122   modifiedEvent._object = object;
123   this.dispatchEvent(modifiedEvent);
124   
125 };
