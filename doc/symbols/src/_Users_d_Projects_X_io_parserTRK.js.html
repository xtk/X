 44 /**
 45  * Create a parser for the binary .TRK format.
 46  * 
 47  * @constructor
 48  * @extends X.parser
 49  */
 50 X.parserTRK = function() {
 51 
 52   //
 53   // call the standard constructor of X.base
 54   goog.base(this);
 55   
 56   //
 57   // class attributes
 58   
 59   /**
 60    * @inheritDoc
 61    * @const
 62    */
 63   this['_className'] = 'parserTRK';
 64   
 65 };
 66 // inherit from X.parser
 67 goog.inherits(X.parserTRK, X.parser);
 68 
 69 
 70 /**
 71  * @inheritDoc
 72  */
 73 X.parserTRK.prototype.parse = function(object, data) {
 74 
 75   var p = object.points();
 76   var n = object.normals();
 77   var c = object.colors();
 78   
 79   var offset = 0;
 80   
 81   // parse the header of the .TRK file
 82   // Documented here: http://trackvis.org/docs/?subsect=fileformat
 83   var header = {
 84     'id_string': this.parseString(data, offset, 6),
 85     'dim': this.parseUInt16Array(data, offset + 6, 3),
 86     'voxel_size': this.parseFloat32Array(data, offset + 12, 3),
 87     'origin': this.parseFloat32Array(data, offset + 24, 3),
 88     'n_scalars': this.parseUInt16(data, offset + 36),
 89     'scalar_name': this.parseString(data, offset + 38, 200),
 90     'n_properties': this.parseUInt16(data, offset + 238),
 91     'property_name': this.parseString(data, offset + 240, 200),
 92     'vox_to_ras': this.parseFloat32Array(data, offset + 440, 16),
 93     'reserved': this.parseString(data, offset + 504, 444),
 94     'voxel_order': this.parseString(data, offset + 948, 4),
 95     'pad2': this.parseString(data, offset + 952, 4),
 96     'image_orientation_patient': this.parseFloat32Array(data, offset + 956, 6),
 97     'pad1': this.parseString(data, offset + 980, 2),
 98     'invert_x': this.parseUChar8(data, offset + 982),
 99     'invert_y': this.parseUChar8(data, offset + 983),
100     'invert_z': this.parseUChar8(data, offset + 984),
101     'swap_xy': this.parseUChar8(data, offset + 985),
102     'swap_yz': this.parseUChar8(data, offset + 986),
103     'swap_zx': this.parseUChar8(data, offset + 987),
104     'n_count': this.parseUInt32(data, offset + 988),
105     'version': this.parseUInt32(data, offset + 992),
106     'hdr_size': this.parseUInt32(data, offset + 996)
107   };
108   
109   //
110   // parse the data
111   offset = header.hdr_size;
112   
113   var numberOfFibers = header.n_count;
114   
115   // loop through all fibers
116   var fibers = [];
117   
118   var minX = null;
119   var maxX = null;
120   var minY = null;
121   var maxY = null;
122   var minZ = null;
123   var maxZ = null;
124   
125   var i;
126   for (i = 0; i < numberOfFibers; i++) {
127     var numPoints = this.parseUInt32(data, offset);
128     
129     var currentPoints = new X.triplets();
130     
131     offset += 4;
132     
133     // loop through the points of this fiber
134     for ( var j = 0; j < numPoints; j++) {
135       
136       // read coordinates
137       var x = this.parseFloat32(data, offset);
138       offset += 4;
139       
140       var y = this.parseFloat32(data, offset);
141       offset += 4;
142       
143       var z = this.parseFloat32(data, offset);
144       offset += 4;
145       
146       // read scalars
147       var scalars = null;
148       if (header.n_scalars > 0) {
149         scalars = this.parseFloat32Array(data, offset, header.n_scalars);
150         offset += (header.n_scalars * 4);
151       }
152       
153       // Convert coordinates to world space by dividing by spacing
154       x = x / header.voxel_size[0];
155       y = y / header.voxel_size[1];
156       z = z / header.voxel_size[2];
157       
158       currentPoints.add(x, y, z);
159       
160     }
161     
162     // read additional properties, if existing
163     // we don't support them right now in XTK
164     if (header.n_properties > 0) {
165       // var properties = this
166       // .parseFloat32Array(data, offset, header.n_properties);
167       offset += (header.n_properties * 4);
168     }
169     
170     // calculate track length
171     // we don't support it right now in XTK
172     // var length = 0.0;
173     // for (j = 0; j < numPoints - 1; j++) {
174     // length += Math.sqrt(Math.pow(points[j + 1].position[0] -
175     // points[j].position[0], 2) +
176     // Math.pow(points[j + 1].position[1] - points[j].position[1], 2) +
177     // Math.pow(points[j + 1].position[2] - points[j].position[2], 2));
178     // }
179     
180     // we need to get the bounding box of the whole .trk file before we add the
181     // points to properly setup normals
182     
183     var cMinX = currentPoints.minA();
184     var cMaxX = currentPoints.maxA();
185     var cMinY = currentPoints.minB();
186     var cMaxY = currentPoints.maxB();
187     var cMinZ = currentPoints.minC();
188     var cMaxZ = currentPoints.maxC();
189     
190     if (!minX || cMinX < minX) {
191       minX = cMinX;
192     }
193     if (!maxX || cMaxX > maxX) {
194       maxX = cMaxX;
195     }
196     if (!minY || cMinY < minY) {
197       minY = cMinY;
198     }
199     if (!maxY || cMaxY > maxY) {
200       maxY = cMaxY;
201     }
202     if (!minZ || cMinZ < minZ) {
203       minZ = cMinZ;
204     }
205     if (!maxZ || cMaxZ > maxZ) {
206       maxZ = cMaxZ;
207     }
208     
209     // append this track to our fibers list
210     fibers.push(currentPoints);
211     
212   } // end of loop through all tracks
213   
214   // calculate the center based on the bounding box of the whole .trk file
215   var centerX = (minX + maxX) / 2;
216   var centerY = (minY + maxY) / 2;
217   var centerZ = (minZ + maxZ) / 2;
218   
219   // now we have a list of fibers
220   for (i = 0; i < numberOfFibers; i++) {
221     
222     // grab the current points of this fiber
223     var points = fibers[i];
224     var numberOfPoints = points.count();
225     
226     for ( var j = 0; j < numberOfPoints - 1; j++) {
227       
228       // TODO min max check?
229       
230       // grab the point with the currentIndex
231       var currentPoint = points.get(j);
232       // and connect it to the next one
233       var nextPoint = points.get(j + 1);
234       
235       // .. and add both
236       p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
237       p.add(nextPoint[0], nextPoint[1], nextPoint[2]);
238       
239       // calculate and add the normals for the two points
240       var nCurrentPointX = currentPoint[0] - centerX;
241       var nCurrentPointY = currentPoint[1] - centerY;
242       var nCurrentPointZ = currentPoint[2] - centerZ;
243       var nCurrentPointLength = Math.sqrt(nCurrentPointX * nCurrentPointX +
244           nCurrentPointY * nCurrentPointY + nCurrentPointZ * nCurrentPointZ);
245       var nNextPointX = nextPoint[0] - centerX;
246       var nNextPointY = nextPoint[1] - centerY;
247       var nNextPointZ = nextPoint[2] - centerZ;
248       var nNextPointLength = Math.sqrt(nNextPointX * nNextPointX + nNextPointY *
249           nNextPointY + nNextPointZ * nNextPointZ);
250       
251       n.add(nCurrentPointX / nCurrentPointLength, nCurrentPointY /
252           nCurrentPointLength, nCurrentPointZ / nCurrentPointLength);
253       n.add(nNextPointX / nNextPointLength, nNextPointY / nNextPointLength,
254           nNextPointZ / nNextPointLength);
255       
256       var start = currentPoint;
257       var end = nextPoint;
258       
259       var diff = [Math.abs(end[0] - start[0]), Math.abs(end[1] - start[1]),
260                   Math.abs(end[2] - start[2])];
261       
262       var distance = Math.sqrt(diff[0] * diff[0] + diff[1] * diff[1] + diff[2] *
263           diff[2]);
264       
265       diff[0] /= distance;
266       diff[1] /= distance;
267       diff[2] /= distance;
268       
269       // add the color
270       c.add(diff[0], diff[1], diff[2]);
271       c.add(diff[0], diff[1], diff[2]);
272       
273     } // loop through points
274     
275   } // loop through fibers
276   
277   // set the object type to LINES
278   object.setType(X.object.types.LINES);
279   
280   var modifiedEvent = new X.event.ModifiedEvent();
281   modifiedEvent._object = object;
282   this.dispatchEvent(modifiedEvent);
283   
284 };
