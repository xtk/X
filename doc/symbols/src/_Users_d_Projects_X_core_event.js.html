 51 /**
 52  * The superclass class for all events in XTK.
 53  * 
 54  * @constructor
 55  * @param {string} type A type identifier for this event.
 56  * @extends goog.events.Event
 57  */
 58 X.event = function(type) {
 59 
 60   //
 61   // call the default event constructor
 62   goog.base(this, type);
 63   
 64   //
 65   // class attributes
 66   
 67   /**
 68    * The className of this class.
 69    * 
 70    * @type {string}
 71    * @protected
 72    */
 73   this._className = 'event';
 74   
 75 };
 76 // inherit from goog.events.Event
 77 goog.inherits(X.event, goog.events.Event);

 80 /**
 81  * Creates a unique event id.
 82  * 
 83  * @param {string} id The id.
 84  * @return {string} A unique id.
 85  */
 86 X.event.uniqueId = function(id) {
 87 
 88   return goog.events.getUniqueId(id);
 89   
 90 };

 92 /**
 93  * The events of this class.
 94  * 
 95  * @enum {string}
 96  */
 97 X.event.events = {
 98   // the pan event, where the event and focus get moved accordingly
 99   PAN: X.event.uniqueId('pan'),
100   
101   // the rotate event, where only the event gets moved
102   ROTATE: X.event.uniqueId('rotate'),
103   
104   // the zoom event, where the event Z coordinate changes
105   ZOOM: X.event.uniqueId('zoom'),
106   
107   // the render event
108   RENDER: X.event.uniqueId('render'),
109   
110   // the reset view event
111   RESETVIEW: X.event.uniqueId('resetview'),
112   
113   // the object modified event
114   MODIFIED: X.event.uniqueId('modified'),
115   
116   // the loading progress event
117   PROGRESS: X.event.uniqueId('progress'),
118   
119   // the hover event
120   HOVER: X.event.uniqueId('hover'),
121   
122   // the hover end event
123   HOVER_END: X.event.uniqueId('hover_end')
124 
125 };

128 /**
129  * The pan event to initiate moving the event and the focus.
130  * 
131  * @constructor
132  * @extends X.event
133  */
134 X.event.PanEvent = function() {
135 
136   // call the default event constructor
137   goog.base(this, X.event.events.PAN);
138   
139   /**
140    * The distance to pan in screen space.
141    * 
142    * @type {!goog.math.Vec2}
143    * @protected
144    */
145   this._distance = new goog.math.Vec2(0, 0);
146   
147 };

184 /**
185  * The zoom event to initiate zoom in or zoom out.
186  * 
187  * @constructor
188  * @extends X.event
189  */
190 X.event.ZoomEvent = function() {
191 
192   // call the default event constructor
193   goog.base(this, X.event.events.ZOOM);
194   
195   /**
196    * The flag for the zooming direction. If TRUE, the zoom operation will move
197    * the objects closer to the event. If FALSE, further away from the event.
198    * 
199    * @type {!boolean}
200    * @protected
201    */
202   this._in = false;
203   
204   /**
205    * The flag for the zooming speed. If TRUE, the zoom operation will happen
206    * fast. If FALSE, there will be a fine zoom operation.
207    * 
208    * @type {!boolean}
209    * @protected
210    */
211   this._fast = false;
212   
213 };
214 // inherit from goog.events.Event
215 goog.inherits(X.event.ZoomEvent, X.event);
