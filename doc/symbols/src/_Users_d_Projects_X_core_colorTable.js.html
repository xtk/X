 30 // provides
 31 goog.provide('X.colorTable');
 32 
 33 // requires
 34 goog.require('X.base');
 35 goog.require('X.file');
 36 goog.require('goog.structs.Map');
 37 
 38 
 39 
 40 /**
 41  * Create a container for a colorTable.
 42  * 
 43  * @constructor
 44  * @extends X.base
 45  */
 46 X.colorTable = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this['_className'] = 'colorTable';
 60   
 61   /**
 62    * The internal hash map to store the value-color mapping.
 63    * 
 64    * @type {!goog.structs.Map}
 65    */
 66   this._map = new goog.structs.Map();
 67   
 68   // the global id counter
 69   var counter = window["X.Counter"];
 70   // ..get a new unique id
 71   counter.increment();
 72   
 73   /**
 74    * The uniqueId of this object. Each object in XTK has a uniqueId.
 75    * 
 76    * @type {number}
 77    * @protected
 78    */
 79   this['_id'] = counter.value();
 80   
 81   this._file = null;
 82   
 83 };
 84 // inherit from X.base
 85 goog.inherits(X.colorTable, X.base);
 86 
 87 
 88 /**
 89  * Add an entry to this color table.
 90  * 
 91  * @param value
 92  * @param label
 93  * @param r
 94  * @param g
 95  * @param b
 96  * @param a
 97  * @throws {Error} If the given values are invalid.
 98  */
 99 X.colorTable.prototype.add = function(value, label, r, g, b, a) {
100 
101   if (!goog.isNumber(value) || !goog.isNumber(r) || !goog.isNumber(g) ||
102       !goog.isNumber(b) || !goog.isNumber(a)) {
103     
104     throw new Error('Invalid color table entry.');
105     
106   }
107   
108   this._map.set(value, [label, r, g, b, a]);
109   
110   this._dirty = true;
111   
112 };
113 
114 
115 /**
116  * Get the id of this color table.
117  * 
118  * @return {number} The id of this color table.
119  */
120 X.colorTable.prototype.id = function() {
121 
122   return this['_id'];
123   
124 };
125 
126 
127 /**
128  * Get the file of this color table.
129  * 
130  * @return {?X.file} The file of this color table.
131  */
132 X.colorTable.prototype.file = function() {
133 
134   return this._file;
135   
136 };
137 
138 
139 /**
140  * Set the file for this color table.
141  * 
142  * @param {?X.file|string} file The file path or an X.file object containing the
143  *          path.
144  */
145 X.colorTable.prototype.setFile = function(file) {
146 
147   if (!goog.isDefAndNotNull(file)) {
148     
149     // null files are allowed
150     this._file = null;
151     return;
152     
153   }
154   
155   if (goog.isString(file)) {
156     
157     file = new X.file(file);
158     
159   }
160   
161   this._file = file;
162   
163 };
