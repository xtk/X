 15 /**
 16  * Create a wrapper around a javascript console with convenience methods for
 17  * outputting information.
 18  *
 19  * @constructor
 20  * @extends {X.base}
 21  */
 22 X.console = function() {
 23 
 24   // call the standard constructor of X.base
 25   goog.base(this);
 26 
 27   //
 28   // class attributes
 29 
 30   /**
 31    * @inheritDoc
 32    * @const
 33    */
 34   this._className = 'console';
 35 
 36   /**
 37    * The wrapped console of the browser.
 38    *
 39    * @type {?Element}
 40    * @private
 41    */
 42   this._console_ = null;
 43 
 44 };
 45 // inherit from X.base
 46 goog.inherits(X.console, X.base);
 47 
 48 
 49 /**
 50  * Returns the wrapped console of the browser. If it does not exist, set up the
 51  * wrapping first.
 52  *
 53  * @return {!Element} The wrapped console of the browser.
 54  * @throws {X.exception} An exception if the wrapping failed.
 55  * @private
 56  */
 57 // why "_" ?
 58 X.console.prototype.getConsole_ = function() {
 59 
 60   if (!this._console_) {
 61 
 62     // grab the console if available
 63     var _window = goog.dom.getWindow(goog.dom.getDocument());
 64 
 65     if (!_window) {
 66 
 67       // could not get the window
 68       throw new X.exception('Fatal: Could not get the window!');
 69 
 70     }
 71 
 72     var _console = _window.console;
 73 
 74     if (!_console) {
 75 
 76       // could not get the console
 77       throw new X.exception('Fatal: The console is not available!');
 78 
 79     }
 80 
 81     this._console_ = _console;
 82 
 83   }
 84 
 85   return this._console_;
 86 
 87 };

 89 
 90 /**
 91  * Prints a message to the console.
 92  *
 93  * @param {?string} message The message to print.
 94  */
 95 X.console.prototype.out = function(message) {
 96 
 97   var _console = this.getConsole_();
 98 
 99   if (_console) {
100 
101     _console.log(message);
102 
103   }
104 
105 };
106 
107 
108 /**
109  * Prints an error message to the console.
110  *
111  * @param {?string} errorMessage The message to print.
112  */
113 X.console.prototype.err = function(errorMessage) {
114 
115   var _console = this.getConsole_();
116 
117   if (_console) {
118 
119     _console.error(errorMessage);
120 
121   }
122 
123 };
